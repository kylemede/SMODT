36 /***********************************************************************
37 Log factorial function
38 ***********************************************************************/
39 double LnFac(int32_t n) {
40  // log factorial function. gives natural logarithm of n!
41 
42  // define constants
43  static const double // coefficients in Stirling approximation
44  C0 = 0.918938533204672722, // ln(sqrt(2*pi))
45  C1 = 1./12.,
46  C3 = -1./360.;
47  // C5 = 1./1260., // use r^5 term if FAK_LEN < 50
48  // C7 = -1./1680.; // use r^7 term if FAK_LEN < 20
49  // static variables
50  static double fac_table[FAK_LEN]; // table of ln(n!):
51  static int initialized = 0; // remember if fac_table has been initialized
52 
53  if (n < FAK_LEN) {
54  if (n <= 1) {
55  if (n < 0) FatalError("Parameter negative in LnFac function");
56  return 0;
57  }
58  if (!initialized) { // first time. Must initialize table
59  // make table of ln(n!)
60  double sum = fac_table[0] = 0.;
61  for (int i=1; i<FAK_LEN; i++) {
62  sum += log(double(i));
63  fac_table[i] = sum;
64  }
65  initialized = 1;
66  }
67  return fac_table[n];
68  }
69  // not found in table. use Stirling approximation
70  double n1, r;
71  n1 = n; r = 1. / n1;
72  return (n1 + 0.5)*log(n1) - n1 + C0 + r*(C1 + r*r*C3);
73 }

89 /***********************************************************************
90 Hypergeometric distribution
91 ***********************************************************************/
92 int32_t StochasticLib1::Hypergeometric (int32_t n, int32_t m, int32_t N) {
93  /*
94  This function generates a random variate with the hypergeometric
95  distribution. This is the distribution you get when drawing balls without
96  replacement from an urn with two colors. n is the number of balls you take,
97  m is the number of red balls in the urn, N is the total number of balls in
98  the urn, and the return value is the number of red balls you get.
99 
100  This function uses inversion by chop-down search from the mode when
101  parameters are small, and the ratio-of-uniforms method when the former
102  method would be too slow or would give overflow.
103  */
104 
105  int32_t fak, addd; // used for undoing transformations
106  int32_t x; // result
107 
108  // check if parameters are valid
109  if (n > N || m > N || n < 0 || m < 0) {
110  FatalError("Parameter out of range in hypergeometric function");}
111 
112  // symmetry transformations
113  fak = 1; addd = 0;
114  if (m > N/2) {
115  // invert m
116  m = N - m;
117  fak = -1; addd = n;
118  }
119  if (n > N/2) {
120  // invert n
121  n = N - n;
122  addd += fak * m; fak = - fak;
123  }
124  if (n > m) {
125  // swap n and m
126  x = n; n = m; m = x;
127  }
128  // cases with only one possible result end here
129  if (n == 0) return addd;
130 
131  //------------------------------------------------------------------
132  // choose method
133  //------------------------------------------------------------------
134  if (N > 680 || n > 70) {
135  // use ratio-of-uniforms method
136  x = HypRatioOfUnifoms (n, m, N);
137  }
138  else {
139  // inversion method, using chop-down search from mode
140  x = HypInversionMod (n, m, N);
141  }
142  // undo symmetry transformations
143  return x * fak + addd;
144 }

151 int32_t StochasticLib1::HypInversionMod (int32_t n, int32_t m, int32_t N) {
152  /*
153  Subfunction for Hypergeometric distribution. Assumes 0 <= n <= m <= N/2.
154  Overflow protection is needed when N > 680 or n > 75.
155 
156  Hypergeometric distribution by inversion method, using down-up
157  search starting at the mode using the chop-down technique.
158 
159  This method is faster than the rejection method when the variance is low.
160  */
161 
162  // Sampling
163  int32_t I; // Loop counter
164  int32_t L = N - m - n; // Parameter
165  double modef; // mode, float
166  double Mp, np; // m + 1, n + 1
167  double p; // temporary
168  double U; // uniform random
169  double c, d; // factors in iteration
170  double divisor; // divisor, eliminated by scaling
171  double k1, k2; // float version of loop counter
172  double L1 = L; // float version of L
173 
174  Mp = (double)(m + 1);
175  np = (double)(n + 1);
176 
177  if (N != hyp_N_last || m != hyp_m_last || n != hyp_n_last) {
178  // set-up when parameters have changed
179  hyp_N_last = N; hyp_m_last = m; hyp_n_last = n;
180 
181  p = Mp / (N + 2.);
182  modef = np * p; // mode, real
183  hyp_mode = (int32_t)modef; // mode, integer
184  if (hyp_mode == modef && p == 0.5) {
185  hyp_mp = hyp_mode--;
186  }
187  else {
188  hyp_mp = hyp_mode + 1;
189  }
190  // mode probability, using log factorial function
191  // (may read directly from fac_table if N < FAK_LEN)
192  hyp_fm = exp(LnFac(N-m) - LnFac(L+hyp_mode) - LnFac(n-hyp_mode)
193  + LnFac(m) - LnFac(m-hyp_mode) - LnFac(hyp_mode)
194  - LnFac(N) + LnFac(N-n) + LnFac(n) );
195 
196  // safety bound - guarantees at least 17 significant decimal digits
197  // bound = min(n, (int32_t)(modef + k*c'))
198  hyp_bound = (int32_t)(modef + 11. * sqrt(modef * (1.-p) * (1.-n/(double)N)+1.));
199  if (hyp_bound > n) hyp_bound = n;
200  }
201 
202  // loop until accepted
203  while(1) {
204  U = Random(); // uniform random number to be converted
205 
206  // start chop-down search at mode
207  if ((U -= hyp_fm) <= 0.) return(hyp_mode);
208  c = d = hyp_fm;
209 
210  // alternating down- and upward search from the mode
211  k1 = hyp_mp - 1; k2 = hyp_mode + 1;
212  for (I = 1; I <= hyp_mode; I++, k1--, k2++) {
213  // Downward search from k1 = hyp_mp - 1
214  divisor = (np - k1)*(Mp - k1);
215  // Instead of dividing c with divisor, we multiply U and d because
216  // multiplication is faster. This will give overflow if N > 800
217  U *= divisor; d *= divisor;
218  c *= k1 * (L1 + k1);
219  if ((U -= c) <= 0.) return(hyp_mp - I - 1); // = k1 - 1
220 
221  // Upward search from k2 = hyp_mode + 1
222  divisor = k2 * (L1 + k2);
223  // re-scale parameters to avoid time-consuming division
224  U *= divisor; c *= divisor;
225  d *= (np - k2) * (Mp - k2);
226  if ((U -= d) <= 0.) return(hyp_mode + I); // = k2
227  // Values of n > 75 or N > 680 may give overflow if you leave out this..
228  // overflow protection
229  // if (U > 1.E100) {U *= 1.E-100; c *= 1.E-100; d *= 1.E-100;}
230  }
231 
232  // Upward search from k2 = 2*mode + 1 to bound
233  for (k2 = I = hyp_mp + hyp_mode; I <= hyp_bound; I++, k2++) {
234  divisor = k2 * (L1 + k2);
235  U *= divisor;
236  d *= (np - k2) * (Mp - k2);
237  if ((U -= d) <= 0.) return(I);
238  // more overflow protection
239  // if (U > 1.E100) {U *= 1.E-100; d *= 1.E-100;}
240  }
241  }
242 }

245 int32_t StochasticLib1::HypRatioOfUnifoms (int32_t n, int32_t m, int32_t N) {
246  /*
247  Subfunction for Hypergeometric distribution using the ratio-of-uniforms
248  rejection method.
249 
250  This code is valid for 0 < n <= m <= N/2.
251 
252  The computation time hardly depends on the parameters, except that it matters
253  a lot whether parameters are within the range where the LnFac function is
254  tabulated.
255 
256  Reference: E. Stadlober: "The ratio of uniforms approach for generating
257  discrete random variates". Journal of Computational and Applied Mathematics,
258  vol. 31, no. 1, 1990, pp. 181-189.
259  */
260  int32_t L; // N-m-n
261  int32_t mode; // mode
262  int32_t k; // integer sample
263  double x; // real sample
264  double rNN; // 1/(N*(N+2))
265  double my; // mean
266  double var; // variance
267  double u; // uniform random
268  double lf; // ln(f(x))
269 
270  L = N - m - n;
271  if (hyp_N_last != N || hyp_m_last != m || hyp_n_last != n) {
272  hyp_N_last = N; hyp_m_last = m; hyp_n_last = n; // Set-up
273  rNN = 1. / ((double)N*(N+2)); // make two divisions in one
274  my = (double)n * m * rNN * (N+2); // mean = n*m/N
275  mode = (int32_t)(double(n+1) * double(m+1) * rNN * N); // mode = floor((n+1)*(m+1)/(N+2))
276  var = (double)n * m * (N-m) * (N-n) / ((double)N*N*(N-1));// variance
277  hyp_h = sqrt(SHAT1 * (var+0.5)) + SHAT2; // hat width
278  hyp_a = my + 0.5; // hat center
279  hyp_fm = fc_lnpk(mode, L, m, n); // maximum
280  hyp_bound = (int32_t)(hyp_a + 4.0 * hyp_h); // safety-bound
281  if (hyp_bound > n) hyp_bound = n;
282  }
283  while(1) {
284  u = Random(); // uniform random number
285  if (u == 0) continue; // avoid division by 0
286  x = hyp_a + hyp_h * (Random()-0.5) / u; // generate hat distribution
287  if (x < 0. || x > 2E9) continue; // reject, avoid overflow
288  k = (int32_t)x;
289  if (k > hyp_bound) continue; // reject if outside range
290  lf = hyp_fm - fc_lnpk(k,L,m,n); // ln(f(k))
291  if (u * (4.0 - u) - 3.0 <= lf) break; // lower squeeze accept
292  if (u * (u-lf) > 1.0) continue; // upper squeeze reject
293  if (2.0 * log(u) <= lf) break; // final acceptance
294  }
295  return k;
296 }

307 /***********************************************************************
308 Multivariate hypergeometric distribution
309 ***********************************************************************/
310 void StochasticLib1::MultiHypergeometric (int32_t * destination, int32_t * source, int32_t n, int colors) {
311  /*
312  This function generates a vector of random variates, each with the
313  hypergeometric distribution.
314 
315  The multivariate hypergeometric distribution is the distribution you
316  get when drawing balls from an urn with more than two colors, without
317  replacement.
318 
319  Parameters:
320  destination: An output array to receive the number of balls of each
321  color. Must have space for at least 'colors' elements.
322  source: An input array containing the number of balls of each
323  color in the urn. Must have 'colors' elements.
324  All elements must be non-negative.
325  n: The number of balls drawn from the urn.
326  Can't exceed the total number of balls in the urn.
327  colors: The number of possible colors.
328  */
329  int32_t sum, x, y;
330  int i;
331  if (n < 0 || colors < 0) FatalError("Parameter negative in multihypergeo function");
332  if (colors == 0) return;
333 
334  // compute total number of balls
335  for (i=0, sum=0; i<colors; i++) {
336  y = source[i];
337  if (y < 0) FatalError("Parameter negative in multihypergeo function");
338  sum += y;
339  }
340  if (n > sum) FatalError("n > sum in multihypergeo function");
341 
342  for (i=0; i<colors-1; i++) {
343  // generate output by calling hypergeometric colors-1 times
344  y = source[i];
345  x = Hypergeometric(n, y, sum);
346  n -= x; sum -= y;
347  destination[i] = x;
348  }
349  // get the last one
350  destination[i] = n;
351 }

354 /***********************************************************************
355 Poisson distribution
356 ***********************************************************************/
357 int32_t StochasticLib1::Poisson (double L) {
358  /*
359  This function generates a random variate with the poisson distribution.
360 
361  Uses inversion by chop-down method for L < 17, and ratio-of-uniforms
362  method for L >= 17.
363 
364  For L < 1.E-6 numerical inaccuracy is avoided by direct calculation.
365  */
366 
367  //------------------------------------------------------------------
368  // choose method
369  //------------------------------------------------------------------
370  if (L < 17) {
371  if (L < 1.E-6) {
372  if (L == 0) return 0;
373  if (L < 0) FatalError("Parameter negative in poisson function");
374 
375  //--------------------------------------------------------------
376  // calculate probabilities
377  //--------------------------------------------------------------
378  // For extremely small L we calculate the probabilities of x = 1
379  // and x = 2 (ignoring higher x). The reason for using this
380  // method is to prevent numerical inaccuracies in other methods.
381  //--------------------------------------------------------------
382  return PoissonLow(L);
383  }
384  else {
385  //--------------------------------------------------------------
386  // inversion method
387  //--------------------------------------------------------------
388  // The computation time for this method grows with L.
389  // Gives overflow for L > 80
390  //--------------------------------------------------------------
391  return PoissonInver(L);
392  }
393  }
394  else {
395  if (L > 2.E9) FatalError("Parameter too big in poisson function");
396 
397  //----------------------------------------------------------------
398  // ratio-of-uniforms method
399  //----------------------------------------------------------------
400  // The computation time for this method does not depend on L.
401  // Use where other methods would be slower.
402  //----------------------------------------------------------------
403  return PoissonRatioUniforms(L);
404  }
405 }

409 Subfunctions used by poisson

411 int32_t StochasticLib1::PoissonLow(double L) {
412  /*
413  This subfunction generates a random variate with the poisson
414  distribution for extremely low values of L.
415 
416  The method is a simple calculation of the probabilities of x = 1
417  and x = 2. Higher values are ignored.
418 
419  The reason for using this method is to avoid the numerical inaccuracies
420  in other methods.
421  */
422  double d, r;
423  d = sqrt(L);
424  if (Random() >= d) return 0;
425  r = Random() * d;
426  if (r > L * (1.-L)) return 0;
427  if (r > 0.5 * L*L * (1.-L)) return 1;
428  return 2;
429 }

432 int32_t StochasticLib1::PoissonInver(double L) {
433  /*
434  This subfunction generates a random variate with the poisson
435  distribution using inversion by the chop down method (PIN).
436 
437  Execution time grows with L. Gives overflow for L > 80.
438 
439  The value of bound must be adjusted to the maximal value of L.
440  */
441  const int bound = 130; // safety bound. Must be > L + 8*sqrt(L).
442  double r; // uniform random number
443  double f; // function value
444  int32_t x; // return value
445 
446  if (L != pois_L_last) { // set up
447  pois_L_last = L;
448  pois_f0 = exp(-L); // f(0) = probability of x=0
449  }
450  while (1) {
451  r = Random(); x = 0; f = pois_f0;
452  do { // recursive calculation: f(x) = f(x-1) * L / x
453  r -= f;
454  if (r <= 0) return x;
455  x++;
456  f *= L;
457  r *= x; // instead of f /= x
458  }
459  while (x <= bound);
460  }
461 }

464 int32_t StochasticLib1::PoissonRatioUniforms(double L) {
465  /*
466  This subfunction generates a random variate with the poisson
467  distribution using the ratio-of-uniforms rejection method (PRUAt).
468 
469  Execution time does not depend on L, except that it matters whether L
470  is within the range where ln(n!) is tabulated.
471 
472  Reference: E. Stadlober: "The ratio of uniforms approach for generating
473  discrete random variates". Journal of Computational and Applied Mathematics,
474  vol. 31, no. 1, 1990, pp. 181-189.
475  */
476  double u; // uniform random
477  double lf; // ln(f(x))
478  double x; // real sample
479  int32_t k; // integer sample
480 
481  if (pois_L_last != L) {
482  pois_L_last = L; // Set-up
483  pois_a = L + 0.5; // hat center
484  int32_t mode = (int32_t)L; // mode
485  pois_g = log(L);
486  pois_f0 = mode * pois_g - LnFac(mode); // value at mode
487  pois_h = sqrt(SHAT1 * (L+0.5)) + SHAT2; // hat width
488  pois_bound = (int32_t)(pois_a + 6.0 * pois_h); // safety-bound
489  }
490  while(1) {
491  u = Random();
492  if (u == 0) continue; // avoid division by 0
493  x = pois_a + pois_h * (Random() - 0.5) / u;
494  if (x < 0 || x >= pois_bound) continue; // reject if outside valid range
495  k = (int32_t)(x);
496  lf = k * pois_g - LnFac(k) - pois_f0;
497  if (lf >= u * (4.0 - u) - 3.0) break; // quick acceptance
498  if (u * (u - lf) > 1.0) continue; // quick rejection
499  if (2.0 * log(u) <= lf) break; // final acceptance
500  }
501  return k;
502 }

505 /***********************************************************************
506 Binomial distribution
507 ***********************************************************************/
508 int32_t StochasticLib1::Binomial (int32_t n, double p) {
509  /*
510  This function generates a random variate with the binomial distribution.
511 
512  Uses inversion by chop-down method for n*p < 35, and ratio-of-uniforms
513  method for n*p >= 35.
514 
515  For n*p < 1.E-6 numerical inaccuracy is avoided by poisson approximation.
516  */
517  int inv = 0; // invert
518  int32_t x; // result
519  double np = n * p;
520 
521  if (p > 0.5) { // faster calculation by inversion
522  p = 1. - p; inv = 1;
523  }
524  if (n <= 0 || p <= 0) {
525  if (n == 0 || p == 0) {
526  return inv * n; // only one possible result
527  }
528  // error exit
529  FatalError("Parameter out of range in binomial function");
530  }
531 
532  //------------------------------------------------------------------
533  // choose method
534  //------------------------------------------------------------------
535  if (np < 35.) {
536  if (np < 1.E-6) {
537  // Poisson approximation for extremely low np
538  x = PoissonLow(np);
539  }
540  else {
541  // inversion method, using chop-down search from 0
542  x = BinomialInver(n, p);
543  }
544  }
545  else {
546  // ratio of uniforms method
547  x = BinomialRatioOfUniforms(n, p);
548  }
549  if (inv) {
550  x = n - x; // undo inversion
551  }
552  return x;
553 }

560 int32_t StochasticLib1::BinomialInver (int32_t n, double p) {
561  /*
562  Subfunction for Binomial distribution. Assumes p < 0.5.
563 
564  Uses inversion method by search starting at 0.
565 
566  Gives overflow for n*p > 60.
567 
568  This method is fast when n*p is low.
569  */
570  double f0, f, q;
571  int32_t bound;
572  double pn, r, rc;
573  int32_t x, n1, i;
574 
575  // f(0) = probability of x=0 is (1-p)^n
576  // fast calculation of (1-p)^n
577  f0 = 1.; pn = 1.-p; n1 = n;
578  while (n1) {
579  if (n1 & 1) f0 *= pn;
580  pn *= pn; n1 >>= 1;
581  }
582  // calculate safety bound
583  rc = (n + 1) * p;
584  bound = (int32_t)(rc + 11.0*(sqrt(rc) + 1.0));
585  if (bound > n) bound = n;
586  q = p / (1. - p);
587 
588  while (1) {
589  r = Random();
590  // recursive calculation: f(x) = f(x-1) * (n-x+1)/x*p/(1-p)
591  f = f0; x = 0; i = n;
592  do {
593  r -= f;
594  if (r <= 0) return x;
595  x++;
596  f *= q * i;
597  r *= x; // it is faster to multiply r by x than dividing f by x
598  i--;
599  }
600  while (x <= bound);
601  }
602 }

605 int32_t StochasticLib1::BinomialRatioOfUniforms (int32_t n, double p) {
606  /*
607  Subfunction for Binomial distribution. Assumes p < 0.5.
608 
609  Uses the Ratio-of-Uniforms rejection method.
610 
611  The computation time hardly depends on the parameters, except that it matters
612  a lot whether parameters are within the range where the LnFac function is
613  tabulated.
614 
615  Reference: E. Stadlober: "The ratio of uniforms approach for generating
616  discrete random variates". Journal of Computational and Applied Mathematics,
617  vol. 31, no. 1, 1990, pp. 181-189.
618  */
619  double u; // uniform random
620  double q1; // 1-p
621  double np; // n*p
622  double var; // variance
623  double lf; // ln(f(x))
624  double x; // real sample
625  int32_t k; // integer sample
626 
627  if(bino_n_last != n || bino_p_last != p) { // Set_up
628  bino_n_last = n;
629  bino_p_last = p;
630  q1 = 1.0 - p;
631  np = n * p;
632  bino_mode = (int32_t)(np + p); // mode
633  bino_a = np + 0.5; // hat center
634  bino_r1 = log(p / q1);
635  bino_g = LnFac(bino_mode) + LnFac(n-bino_mode);
636  var = np * q1; // variance
637  bino_h = sqrt(SHAT1 * (var+0.5)) + SHAT2; // hat width
638  bino_bound = (int32_t)(bino_a + 6.0 * bino_h);// safety-bound
639  if (bino_bound > n) bino_bound = n; // safety-bound
640  }
641 
642  while (1) { // rejection loop
643  u = Random();
644  if (u == 0) continue; // avoid division by 0
645  x = bino_a + bino_h * (Random() - 0.5) / u;
646  if (x < 0. || x > bino_bound) continue; // reject, avoid overflow
647  k = (int32_t)x; // truncate
648  lf = (k-bino_mode)*bino_r1+bino_g-LnFac(k)-LnFac(n-k);// ln(f(k))
649  if (u * (4.0 - u) - 3.0 <= lf) break; // lower squeeze accept
650  if (u * (u - lf) > 1.0) continue; // upper squeeze reject
651  if (2.0 * log(u) <= lf) break; // final acceptance
652  }
653  return k;
654 }

657 /***********************************************************************
658 Multinomial distribution
659 ***********************************************************************/
660 void StochasticLib1::Multinomial (int32_t * destination, double * source, int32_t n, int colors) {
661  /*
662  This function generates a vector of random variates, each with the
663  binomial distribution.
664 
665  The multinomial distribution is the distribution you get when drawing
666  balls from an urn with more than two colors, with replacement.
667 
668  Parameters:
669  destination: An output array to receive the number of balls of each
670  color. Must have space for at least 'colors' elements.
671  source: An input array containing the probability or fraction
672  of each color in the urn. Must have 'colors' elements.
673  All elements must be non-negative. The sum doesn't have
674  to be 1, but the sum must be positive.
675  n: The number of balls drawn from the urn.
676  colors: The number of possible colors.
677  */
678  double s, sum;
679  int32_t x;
680  int i;
681  if (n < 0 || colors < 0) FatalError("Parameter negative in multinomial function");
682  if (colors == 0) return;
683 
684  // compute sum of probabilities
685  for (i=0, sum=0; i<colors; i++) {
686  s = source[i];
687  if (s < 0) FatalError("Parameter negative in multinomial function");
688  sum += s;
689  }
690  if (sum == 0 && n > 0) FatalError("Zero sum in multinomial function");
691 
692  for (i=0; i<colors-1; i++) {
693  // generate output by calling binomial (colors-1) times
694  s = source[i];
695  if (sum <= s) {
696  // this fixes two problems:
697  // 1. prevent division by 0 when sum = 0
698  // 2. prevent s/sum getting bigger than 1 in case of rounding errors
699  x = n;
700  }
701  else {
702  x = Binomial(n, s/sum);
703  }
704  n -= x; sum -= s;
705  destination[i] = x;
706  }
707  // get the last one
708  destination[i] = n;
709 }

746 double StochasticLib1::Normal(double m, double s) {
747  // normal distribution with mean m and standard deviation s
748  double normal_x1; // first random coordinate (normal_x2 is member of class)
749  double w; // radius
750  if (normal_x2_valid) { // we have a valid result from last call
751  normal_x2_valid = 0;
752  return normal_x2 * s + m;
753  }
754  // make two normally distributed variates by Box-Muller transformation
755  do {
756  normal_x1 = 2. * Random() - 1.;
757  normal_x2 = 2. * Random() - 1.;
758  w = normal_x1*normal_x1 + normal_x2*normal_x2;
759  }
760  while (w >= 1. || w < 1E-30);
761  w = sqrt(log(w)*(-2./w));
762  normal_x1 *= w; normal_x2 *= w; // normal_x1 and normal_x2 are independent normally distributed variates
763  normal_x2_valid = 1; // save normal_x2 for next call
764  return normal_x1 * s + m;
765 }

791 /***********************************************************************
792 Shuffle function
793 ***********************************************************************/
794 void StochasticLib1::Shuffle(int * list, int min, int n) {
795  /*
796  This function makes a list of the n numbers from min to min+n-1
797  in random order.
798 
799  The parameter 'list' must be an array with at least n elements.
800  The array index goes from 0 to n-1.
801 
802  If you want to shuffle something else than integers then use the
803  integers in list as an index into a table of the items you want to shuffle.
804  */
805 
806  int i, j, swap;
807  // put numbers from min to min+n-1 into list
808  for (i=0, j=min; i<n; i++, j++) list[i] = j;
809  // shuffle list
810  for (i=0; i<n-1; i++) {
811  // item number i has n-i numbers to choose between
812  j = IRandom(i,n-1);
813  // swap items i and j
814  swap = list[j]; list[j] = list[i]; list[i] = swap;
815  }
816 }
