30  Poisson distribution
31 ***********************************************************************/
32 int32_t StochasticLib2::Poisson (double L) {
33 /*
34  This function generates a random variate with the poisson distribution.
35 
36  Uses down/up search from the mode by chop-down technique for L < 20,
37  and patchwork rejection method for L >= 20.
38 
39  For L < 1.E-6 numerical inaccuracy is avoided by direct calculation.
40 */
41  //------------------------------------------------------------------
42  // choose method
43  //------------------------------------------------------------------
44  if (L < 20) {
45  if (L < 1.E-6) {
46  if (L == 0) return 0;
47  if (L < 0) FatalError("Parameter negative in poisson function");
48 
49  //--------------------------------------------------------------
50  // calculate probabilities
51  //--------------------------------------------------------------
52  // For extremely small L we calculate the probabilities of x = 1
53  // and x = 2 (ignoring higher x). The reason for using this
54  // method is to prevent numerical inaccuracies in other methods.
55  //--------------------------------------------------------------
56  return PoissonLow(L);}
57 
58  else {
59 
60  //--------------------------------------------------------------
61  // down/up search from mode
62  //--------------------------------------------------------------
63  // The computation time for this method grows with sqrt(L).
64  // Gives overflow for L > 60
65  //--------------------------------------------------------------
66  return PoissonModeSearch(L);}}
67 
68  else {
69  if (L > 2.E9) FatalError("Parameter too big in Poisson function");
70 
71  //----------------------------------------------------------------
72  // patchword rejection method
73  //----------------------------------------------------------------
74  // The computation time for this method does not depend on L.
75  // Use where other methods would be slower.
76  //----------------------------------------------------------------
77  return PoissonPatchwork(L);}}

81  Binomial distributuion
82 ***********************************************************************/
83 int32_t StochasticLib2::Binomial (int32_t n, double p) {
84 /*
85  This function generates a random variate with the binomial distribution.
86 
87  Uses down/up search from the mode by chop-down technique for n*p < 60,
88  and patchwork rejection method for n*p >= 55.
89 
90  For n*p < 1.E-6 numerical inaccuracy is avoided by poisson approximation.
91 */
92  int inv = 0; // invert
93  int32_t x; // result
94  double np = n * p;
95 
96  if (p > 0.5) { // faster calculation by inversion
97  p = 1. - p; inv = 1;}
98 
99  if (n <= 0 || p <= 0) {
100  if (n == 0 || p == 0) {
101  return inv * n; // only one possible result
102  }
103  // error exit
104  FatalError("Parameter negative in binomial function");
105  }
106 
107 
108  //------------------------------------------------------------------
109  // choose method
110  //------------------------------------------------------------------
111  if (np < 55.) {
112  if (np < 1.E-6) {
113  // Poisson approximation for extremely low np
114  x = PoissonLow(np);}
115 
116  else {
117  // inversion method, using chop-down search from 0
118  x = BinomialModeSearch(n, p);
119  }
120  }
121  else {
122  // ratio of uniforms method
123  x = BinomialPatchwork(n, p);
124  }
125  if (inv) {
126  x = n - x; // undo inversion
127  }
128  return x;
129 }

133  Hypergeometric distribution
134 ***********************************************************************/
135 int32_t StochasticLib2::Hypergeometric (int32_t n, int32_t m, int32_t N) {
136 /*
137  This function generates a random variate with the hypergeometric
138  distribution. This is the distribution you get when drawing balls
139  without replacement from an urn with two colors.
140 
141  Uses inversion by chop-down search from the mode when the mean < 20
142  and the patchwork-rejection method when the mean > 20.
143 */
144 
145  int32_t x; // result
146  int32_t fak, addd; // used for undoing transformations
147 
148  // check if parameters are valid
149  if (n > N || m > N || n < 0 || m < 0) {
150  FatalError("Parameter out of range in hypergeometric function");
151  }
152  // transformations
153  fak = 1; addd = 0;
154  if (m > N/2) {
155  // invert m
156  m = N - m;
157  fak = -1; addd = n;
158  }
159  if (n > N/2) {
160  // invert n
161  n = N - n;
162  addd += fak * m; fak = - fak;
163  }
164  if (n > m) {
165  // swap n and m
166  x = n; n = m; m = x;
167  }
168  // cases with only one possible result end here
169  if (n == 0) return addd;
170 
171  //------------------------------------------------------------------
172  // choose method
173  //------------------------------------------------------------------
174  if (double(n) * m >= 20. * N) {
175  // use ratio-of-uniforms method
176  x = HypPatchwork (n, m, N);
177  }
178  else {
179  // inversion method, using chop-down search from mode
180  x = HypInversionMod (n, m, N);
181  }
182  // undo transformations
183  return x * fak + addd;
184 }

191 int32_t StochasticLib2::BinomialModeSearch(int32_t n, double p) {
192 /*
193  Subfunction for Binomial distribution. Assumes p < 0.5.
194 
195  Uses inversion method by down-up search starting at the mode (BMDU).
196 
197  Gives overflow for n*p > 60.
198 
199  This method is fast when n*p is low.
200 */
201  int32_t K, x;
202  double U, c, d, rc, divisor;
203 
204  if (n != bino_n_last || p != bino_p_last) {
205  bino_n_last = n;
206  bino_p_last = p;
207  rc = (n + 1) * p;
208 
209  // safety bound guarantees at least 17 significant decimal digits
210  bino_bound = (int32_t)(rc + 11.0*(sqrt(rc) + 1.0));
211  if (bino_bound > n) bino_bound = n;
212  bino_mode = (int32_t) rc;
213  if (bino_mode == rc && p == 0.5) bino_mode--; // mode
214  bino_r1 = p / (1.0 - p);
215  bino_g = exp(LnFac(n)-LnFac(bino_mode)-LnFac(n-bino_mode) + bino_mode*log(p) + (n-bino_mode)*log(1.-p));
216  }
217  while (1) {
218  U = Random();
219  if ((U -= bino_g) <= 0.0) return(bino_mode);
220  c = d = bino_g;
221 
222  // down- and upward search from the mode
223  for (K = 1; K <= bino_mode; K++) {
224  x = bino_mode - K; // downward search from mode
225  divisor = (n-x) * bino_r1;
226  c *= x + 1;
227  U *= divisor;
228  d *= divisor;
229  if ((U -= c) <= 0.0) return x;
230 
231  x = bino_mode + K; // upward search from mode
232  divisor = x;
233  d *= (n-x+1) * bino_r1;
234  U *= divisor;
235  c *= divisor;
236  if ((U -= d) <= 0.0) return x;}
237 
238  // upward search from 2*mode + 1 to bound
239  for (x = bino_mode + bino_mode + 1; x <= bino_bound; x++) {
240  d *= (n-x+1) * bino_r1;
241  U *= x;
242  if ((U -= d) <= 0.0) return x;
243  }
244  }
245 }

248 int32_t StochasticLib2::BinomialPatchwork(int32_t n, double p) {
249 /*
250  Subfunction for Binomial distribution using the patchwork rejection
251  method (BPRS).
252 */
253 
254  int32_t mode, Dk, X, Y;
255  double nu, q, U, V, W;
256 
257  if (n != bino_n_last || p != bino_p_last) { // set-up
258  bino_n_last = n;
259  bino_p_last = p;
260 
261  nu = (double)(n + 1) * p; q = 1.0 - p; // main parameters
262 
263  // approximate deviation of reflection points k2, k4 from nu - 1/2
264  W = sqrt(nu * q + 0.25);
265 
266  // mode, reflection points k2 and k4, and points k1 and k5, which
267  // delimit the centre region of h(x)
268  mode = (int32_t) nu;
269  Bino_k2 = (int32_t) ceil(nu - 0.5 - W);
270  Bino_k4 = (int32_t) (nu - 0.5 + W);
271  Bino_k1 = Bino_k2 + Bino_k2 - mode + 1;
272  Bino_k5 = Bino_k4 + Bino_k4 - mode;
273 
274  // range width of the critical left and right centre region
275  Bino_dl = (double) (Bino_k2 - Bino_k1);
276  Bino_dr = (double) (Bino_k5 - Bino_k4);
277 
278  // recurrence constants r(k) = p(k)/p(k-1) at k = k1, k2, k4+1, k5+1
279  nu = nu / q; p = p / q;
280  Bino_r1 = nu / (double) Bino_k1 - p; // nu = (n+1)p / q
281  Bino_r2 = nu / (double) Bino_k2 - p; // p = p / q
282  Bino_r4 = nu / (double)(Bino_k4 + 1) - p;
283  Bino_r5 = nu / (double)(Bino_k5 + 1) - p;
284 
285  // reciprocal values of the scale parameters of expon. tail envelopes
286  Bino_ll = log(Bino_r1); // expon. tail left
287  Bino_lr = -log(Bino_r5); // expon. tail right
288 
289  // binomial constants, necessary for computing function values f(k)
290  Bino_l_pq = log(p);
291  Bino_c_pm = mode * Bino_l_pq - LnFac(mode) - LnFac(n - mode);
292 
293  // function values f(k) = p(k)/p(mode) at k = k2, k4, k1, k5
294  Bino_f2 = BinomialF(Bino_k2, n, Bino_l_pq, Bino_c_pm);
295  Bino_f4 = BinomialF(Bino_k4, n, Bino_l_pq, Bino_c_pm);
296  Bino_f1 = BinomialF(Bino_k1, n, Bino_l_pq, Bino_c_pm);
297  Bino_f5 = BinomialF(Bino_k5, n, Bino_l_pq, Bino_c_pm);
298 
299  // area of the two centre and the two exponential tail regions
300  // area of the two immediate acceptance regions between k2, k4
301  Bino_p1 = Bino_f2 * (Bino_dl + 1.); // immed. left
302  Bino_p2 = Bino_f2 * Bino_dl + Bino_p1; // centre left
303  Bino_p3 = Bino_f4 * (Bino_dr + 1.) + Bino_p2; // immed. right
304  Bino_p4 = Bino_f4 * Bino_dr + Bino_p3; // centre right
305  Bino_p5 = Bino_f1 / Bino_ll + Bino_p4; // expon. tail left
306  Bino_p6 = Bino_f5 / Bino_lr + Bino_p5; // expon. tail right
307  }
308 
309  for (;;) {
310  // generate uniform number U -- U(0, p6)
311  // case distinction corresponding to U
312 
313  if ((U = Random() * Bino_p6) < Bino_p2) { // centre left
314  // immediate acceptance region R2 = [k2, mode) *[0, f2), X = k2, ... mode -1
315  if ((V = U - Bino_p1) < 0.) return(Bino_k2 + (int32_t)(U/Bino_f2));
316  // immediate acceptance region R1 = [k1, k2)*[0, f1), X = k1, ... k2-1
317  if ((W = V / Bino_dl) < Bino_f1) return(Bino_k1 + (int32_t)(V/Bino_f1));
318 
319  // computation of candidate X < k2, and its counterpart Y > k2
320  // either squeeze-acceptance of X or acceptance-rejection of Y
321  Dk = (int32_t)(Bino_dl * Random()) + 1;
322  if (W <= Bino_f2 - Dk * (Bino_f2 - Bino_f2/Bino_r2)) { // quick accept of
323  return(Bino_k2 - Dk);} // X = k2 - Dk
324  if ((V = Bino_f2 + Bino_f2 - W) < 1.) { // quick reject of Y
325  Y = Bino_k2 + Dk;
326  if (V <= Bino_f2 + Dk * (1. - Bino_f2)/(Bino_dl + 1.)) { // quick accept of
327  return(Y);} // Y = k2 + Dk
328  if (V <= BinomialF(Y,n,Bino_l_pq,Bino_c_pm)) return(Y);} // final accept of Y
329  X = Bino_k2 - Dk;}
330 
331  else if (U < Bino_p4) { // centre right
332  // immediate acceptance region R3 = [mode, k4+1)*[0, f4), X = mode, ... k4
333  if ((V = U - Bino_p3) < 0.) return(Bino_k4 - (int32_t)((U - Bino_p2)/Bino_f4));
334  // immediate acceptance region R4 = [k4+1, k5+1)*[0, f5)
335  if ((W = V / Bino_dr) < Bino_f5) return(Bino_k5 - (int32_t)(V/Bino_f5));
336 
337  // computation of candidate X > k4, and its counterpart Y < k4
338  // either squeeze-acceptance of X or acceptance-rejection of Y
339  Dk = (int32_t)(Bino_dr * Random()) + 1;
340  if (W <= Bino_f4 - Dk * (Bino_f4 - Bino_f4*Bino_r4)) { // quick accept of
341  return(Bino_k4 + Dk);} // X = k4 + Dk
342  if ((V = Bino_f4 + Bino_f4 - W) < 1.) { // quick reject of Y
343  Y = Bino_k4 - Dk;
344  if (V <= Bino_f4 + Dk * (1. - Bino_f4)/ Bino_dr) { // quick accept of
345  return(Y);} // Y = k4 - Dk
346  if (V <= BinomialF(Y,n,Bino_l_pq,Bino_c_pm)) return(Y); // final accept of Y
347  }
348  X = Bino_k4 + Dk;
349  }
350  else {
351  W = Random();
352  if (U < Bino_p5) { // expon. tail left
353  Dk = (int32_t)(1. - log(W)/Bino_ll);
354  if ((X = Bino_k1 - Dk) < 0) continue; // 0 <= X <= k1 - 1
355  W *= (U - Bino_p4) * Bino_ll; // W -- U(0, h(x))
356  if (W <= Bino_f1 - Dk * (Bino_f1 - Bino_f1/Bino_r1)) {
357  return X; // quick accept of X
358  }
359  }
360  else { // expon. tail right
361  Dk = (int32_t)(1. - log(W)/Bino_lr);
362  if ((X = Bino_k5 + Dk) > n ) continue; // k5 + 1 <= X <= n
363  W *= (U - Bino_p5) * Bino_lr; // W -- U(0, h(x))
364  if (W <= Bino_f5 - Dk * (Bino_f5 - Bino_f5*Bino_r5)) {
365  return X; // quick accept of X
366  }
367  }
368  }
369 
370 
371  // acceptance-rejection test of candidate X from the original area
372  // test, whether W <= BinomialF(k), with W = U*h(x) and U -- U(0, 1)
373  // log BinomialF(X) = (X - mode)*log(p/q) - log X!(n - X)! + log mode!(n - mode)!
374  if (log(W) <= X*Bino_l_pq - LnFac(X) - LnFac(n - X) - Bino_c_pm) {
375  return X;
376  }
377  }
378 }

391 int32_t StochasticLib2::PoissonModeSearch(double L) {
392 /*
393  This subfunction generates a random variate with the poisson
394  distribution by down/up search from the mode, using the chop-down
395  technique (PMDU).
396 
397  Execution time grows with sqrt(L). Gives overflow for L > 60.
398 */
399  double r, c, d;
400  int32_t x, i, mode;
401 
402  mode = (int32_t)L;
403 
404  if (L != pois_L_last) { // set up
405  pois_L_last = L;
406  pois_bound = (int32_t)floor(L+0.5 + 7.0 * (sqrt(L+L+1.) + 1.5));// safety-bound
407  pois_f0 = exp(mode * log(L) - L - LnFac(mode)); // probability of x=mode
408  }
409 
410  while (1) {
411  r = Random();
412  if ((r -= pois_f0) <= 0) return mode;
413  c = d = pois_f0;
414 
415  // alternating down/up search from the mode
416  for (i=1; i<=mode; i++) {
417  // down
418  x = mode - i;
419  c *= x + 1;
420  r *= L; d *= L;
421  if ((r -= c) <= 0) return x;
422  // up
423  x = mode + i;
424  d *= L;
425  r *= x; c *= x;
426  if ((r -= d) <= 0) return x;
427  }
428  // continue upward search from 2*mode+1 to bound
429  for (x = mode + mode + 1; x <= pois_bound; x++) {
430  d *= L;
431  r *= x;
432  if ((r -= d) <= 0) return x;
433  }
434  }
435 }

438 int32_t StochasticLib2::PoissonPatchwork(double L) {
439 /*
440  This subfunction generates a random variate with the poisson
441  distribution using the Patchwork Rejection method (PPRS):
442  The area below the histogram function f(x) is rearranged in
443  its body by two point reflections. Within a large center
444  interval variates are sampled efficiently by rejection from
445  uniform hats. Rectangular immediate acceptance regions speed
446  up the generation. The remaining tails are covered by
447  exponential functions.
448 
449  For detailed explanation, see:
450  Stadlober, E & Zechner, H: "The Patchwork Rejection Technique for
451  Sampling from Unimodal Distributions". ACM Transactions on Modeling
452  and Computer Simulation, vol. 9, no. 1, 1999, p. 59-83.
453 
454  This method is valid for L >= 10.
455 
456  The computation time hardly depends on L, except that it matters
457  a lot whether L is within the range where the LnFac function is
458  tabulated.
459 */
460 
461  int32_t mode, Dk, X, Y;
462  double Ds, U, V, W;
463 
464  if (L != pois_L_last) { // set-up
465  pois_L_last = L;
466 
467  // approximate deviation of reflection points k2, k4 from L - 1/2
468  Ds = sqrt(L + 0.25);
469 
470  // mode, reflection points k2 and k4, and points k1 and k5, which
471  // delimit the centre region of h(x)
472  mode = (int32_t) L;
473  Pois_k2 = (int32_t) ceil(L - 0.5 - Ds);
474  Pois_k4 = (int32_t) (L - 0.5 + Ds);
475  Pois_k1 = Pois_k2 + Pois_k2 - mode + 1;
476  Pois_k5 = Pois_k4 + Pois_k4 - mode;
477 
478  // range width of the critical left and right centre region
479  Pois_dl = (double) (Pois_k2 - Pois_k1);
480  Pois_dr = (double) (Pois_k5 - Pois_k4);
481 
482  // recurrence constants r(k) = p(k)/p(k-1) at k = k1, k2, k4+1, k5+1
483  Pois_r1 = L / (double) Pois_k1;
484  Pois_r2 = L / (double) Pois_k2;
485  Pois_r4 = L / (double)(Pois_k4 + 1);
486  Pois_r5 = L / (double)(Pois_k5 + 1);
487 
488  // reciprocal values of the scale parameters of expon. tail envelopes
489  Pois_ll = log(Pois_r1); // expon. tail left
490  Pois_lr = -log(Pois_r5); // expon. tail right
491 
492  // Poisson constants, necessary for computing function values f(k)
493  Pois_l_my = log(L);
494  Pois_c_pm = mode * Pois_l_my - LnFac(mode);
495 
496  // function values f(k) = p(k)/p(mode) at k = k2, k4, k1, k5
497  Pois_f2 = PoissonF(Pois_k2, Pois_l_my, Pois_c_pm);
498  Pois_f4 = PoissonF(Pois_k4, Pois_l_my, Pois_c_pm);
499  Pois_f1 = PoissonF(Pois_k1, Pois_l_my, Pois_c_pm);
500  Pois_f5 = PoissonF(Pois_k5, Pois_l_my, Pois_c_pm);
501 
502  // area of the two centre and the two exponential tail regions
503  // area of the two immediate acceptance regions between k2, k4
504  Pois_p1 = Pois_f2 * (Pois_dl + 1.); // immed. left
505  Pois_p2 = Pois_f2 * Pois_dl + Pois_p1; // centre left
506  Pois_p3 = Pois_f4 * (Pois_dr + 1.) + Pois_p2; // immed. right
507  Pois_p4 = Pois_f4 * Pois_dr + Pois_p3; // centre right
508  Pois_p5 = Pois_f1 / Pois_ll + Pois_p4; // expon. tail left
509  Pois_p6 = Pois_f5 / Pois_lr + Pois_p5; // expon. tail right
510  }
511 
512  for (;;) {
513  // generate uniform number U -- U(0, p6)
514  // case distinction corresponding to U
515  if ((U = Random() * Pois_p6) < Pois_p2) { // centre left
516 
517  // immediate acceptance region R2 = [k2, mode) *[0, f2), X = k2, ... mode -1
518  if ((V = U - Pois_p1) < 0.0) return(Pois_k2 + (int32_t)(U/Pois_f2));
519  // immediate acceptance region R1 = [k1, k2)*[0, f1), X = k1, ... k2-1
520  if ((W = V / Pois_dl) < Pois_f1 ) return(Pois_k1 + (int32_t)(V/Pois_f1));
521 
522  // computation of candidate X < k2, and its counterpart Y > k2
523  // either squeeze-acceptance of X or acceptance-rejection of Y
524  Dk = (int32_t)(Pois_dl * Random()) + 1;
525  if (W <= Pois_f2 - Dk * (Pois_f2 - Pois_f2/Pois_r2)) { // quick accept of
526  return(Pois_k2 - Dk); // X = k2 - Dk
527  }
528 
529  if ((V = Pois_f2 + Pois_f2 - W) < 1.0) { // quick reject of Y
530  Y = Pois_k2 + Dk;
531  if (V <= Pois_f2 + Dk * (1.0 - Pois_f2)/(Pois_dl + 1.0)) { // quick accept of
532  return Y; // Y = k2 + Dk
533  }
534  if (V <= PoissonF(Y, Pois_l_my, Pois_c_pm)) return Y; // final accept of Y
535  }
536  X = Pois_k2 - Dk;
537  }
538  else if (U < Pois_p4) { // centre right
539  // immediate acceptance region R3 = [mode, k4+1)*[0, f4), X = mode, ... k4
540  if ((V = U - Pois_p3) < 0.) return(Pois_k4 - (int32_t)((U - Pois_p2)/Pois_f4));
541  // immediate acceptance region R4 = [k4+1, k5+1)*[0, f5)
542  if ((W = V / Pois_dr) < Pois_f5) return(Pois_k5 - (int32_t)(V/Pois_f5));
543 
544  // computation of candidate X > k4, and its counterpart Y < k4
545  // either squeeze-acceptance of X or acceptance-rejection of Y
546  Dk = (int32_t)(Pois_dr * Random()) + 1L;
547  if (W <= Pois_f4 - Dk * (Pois_f4 - Pois_f4*Pois_r4)) { // quick accept of
548  return (Pois_k4 + Dk); // X = k4 + Dk
549  }
550  if ((V = Pois_f4 + Pois_f4 - W) < 1.0) { // quick reject of Y
551  Y = Pois_k4 - Dk;
552  if (V <= Pois_f4 + Dk * (1.0 - Pois_f4)/ Pois_dr) { // quick accept of
553  return Y; // Y = k4 - Dk
554  }
555  if (V <= PoissonF(Y, Pois_l_my, Pois_c_pm)) return Y; // final accept of Y
556  }
557  X = Pois_k4 + Dk;
558  }
559  else {
560  W = Random();
561  if (U < Pois_p5) { // expon. tail left
562  Dk = (int32_t)(1.0 - log(W)/Pois_ll);
563  if ((X = Pois_k1 - Dk) < 0L) continue; // 0 <= X <= k1 - 1
564  W *= (U - Pois_p4) * Pois_ll; // W -- U(0, h(x))
565  if (W <= Pois_f1-Dk * (Pois_f1-Pois_f1/Pois_r1)) return X; // quick accept of X
566  }
567  else { // expon. tail right
568  Dk = (int32_t)(1.0 - log(W)/Pois_lr);
569  X = Pois_k5 + Dk; // X >= k5 + 1
570  W *= (U - Pois_p5) * Pois_lr; // W -- U(0, h(x))
571  if (W <= Pois_f5-Dk * (Pois_f5-Pois_f5*Pois_r5)) return X; // quick accept of X
572  }
573  }
574 
575  // acceptance-rejection test of candidate X from the original area
576  // test, whether W <= f(k), with W = U*h(x) and U -- U(0, 1)
577  // log f(X) = (X - mode)*log(L) - log X! + log mode!
578  if (log(W) <= X * Pois_l_my - LnFac(X) - Pois_c_pm) return X;
579  }
580 }

593 int32_t StochasticLib2::HypPatchwork (int32_t n, int32_t m, int32_t N) {
594 /*
595  Subfunction for Hypergeometric distribution.
596 
597  This method is valid only for mode >= 10 and 0 <= n <= m <= N/2.
598 
599  This method is fast when called repeatedly with the same parameters, but
600  slow when the parameters change due to a high setup time. The computation
601  time hardly depends on the parameters, except that it matters a lot whether
602  parameters are within the range where the LnFac function is tabulated.
603 
604  Uses the Patchwork Rejection method of Heinz Zechner (HPRS).
605  The area below the histogram function f(x) in its body is rearranged by
606  two point reflections. Within a large center interval variates are sampled
607  efficiently by rejection from uniform hats. Rectangular immediate acceptance
608  regions speed up the generation. The remaining tails are covered by
609  exponential functions.
610 
611  For detailed explanation, see:
612  Stadlober, E & Zechner, H: "The Patchwork Rejection Technique for
613  Sampling from Unimodal Distributions". ACM Transactions on Modeling
614  and Computer Simulation, vol. 9, no. 1, 1999, p. 59-83.
615 
616 */
617 
618  int32_t mode, Dk, X, V;
619  double Mp, np, p, modef, U, Y, W; // (X, Y) <-> (V, W)
620 
621  if (N != hyp_N_last || m != hyp_m_last || n != hyp_n_last) {
622  // set-up when parameters have changed
623  hyp_N_last = N; hyp_m_last = m; hyp_n_last = n;
624 
625  Mp = (double)(m + 1);
626  np = (double)(n + 1);
627  Hyp_L = N - m - n;
628 
629  p = Mp / (N + 2.);
630  modef = np * p;
631 
632  // approximate deviation of reflection points k2, k4 from modef - 1/2
633  U = sqrt(modef * (1. - p) * (1. - (n + 2.)/(N + 3.)) + 0.25);
634 
635  // mode, reflection points k2 and k4, and points k1 and k5, which
636  // delimit the centre region of h(x)
637  // k2 = ceil (modef - 1/2 - U), k1 = 2*k2 - (mode - 1 + delta_ml)
638  // k4 = floor(modef - 1/2 + U), k5 = 2*k4 - (mode + 1 - delta_mr)
639  mode = (int32_t)modef;
640  Hyp_k2 = (int32_t)ceil(modef - 0.5 - U);
641  if (Hyp_k2 >= mode) Hyp_k2 = mode - 1;
642  Hyp_k4 = (int32_t)(modef - 0.5 + U);
643  Hyp_k1 = Hyp_k2 + Hyp_k2 - mode + 1; // delta_ml = 0
644  Hyp_k5 = Hyp_k4 + Hyp_k4 - mode; // delta_mr = 1
645 
646  // range width of the critical left and right centre region
647  Hyp_dl = (double) (Hyp_k2 - Hyp_k1);
648  Hyp_dr = (double) (Hyp_k5 - Hyp_k4);
649 
650  // recurrence constants r(k) = p(k)/p(k-1) at k = k1, k2, k4+1, k5+1
651  Hyp_r1 = (np/(double) Hyp_k1 - 1.) * (Mp - Hyp_k1)/(double)(Hyp_L + Hyp_k1);
652  Hyp_r2 = (np/(double) Hyp_k2 - 1.) * (Mp - Hyp_k2)/(double)(Hyp_L + Hyp_k2);
653  Hyp_r4 = (np/(double)(Hyp_k4+1) - 1.) * (m - Hyp_k4)/(double)(Hyp_L + Hyp_k4 + 1);
654  Hyp_r5 = (np/(double)(Hyp_k5+1) - 1.) * (m - Hyp_k5)/(double)(Hyp_L + Hyp_k5 + 1);
655 
656  // reciprocal values of the scale parameters of expon. tail envelopes
657  Hyp_ll = log(Hyp_r1); // expon. tail left
658  Hyp_lr = -log(Hyp_r5); // expon. tail right
659 
660  // hypergeom. constant, necessary for computing function values f(k)
661  Hyp_c_pm = fc_lnpk(mode, Hyp_L, m, n);
662 
663  // function values f(k) = p(k)/p(mode) at k = k2, k4, k1, k5
664  Hyp_f2 = exp(Hyp_c_pm - fc_lnpk(Hyp_k2, Hyp_L, m, n));
665  Hyp_f4 = exp(Hyp_c_pm - fc_lnpk(Hyp_k4, Hyp_L, m, n));
666  Hyp_f1 = exp(Hyp_c_pm - fc_lnpk(Hyp_k1, Hyp_L, m, n));
667  Hyp_f5 = exp(Hyp_c_pm - fc_lnpk(Hyp_k5, Hyp_L, m, n));
668 
669  // area of the two centre and the two exponential tail regions
670  // area of the two immediate acceptance regions between k2, k4
671  Hyp_p1 = Hyp_f2 * (Hyp_dl+1.); // immed. left
672  Hyp_p2 = Hyp_f2 * Hyp_dl + Hyp_p1; // centre left
673  Hyp_p3 = Hyp_f4 * (Hyp_dr+1.) + Hyp_p2; // immed. right
674  Hyp_p4 = Hyp_f4 * Hyp_dr + Hyp_p3; // centre right
675  Hyp_p5 = Hyp_f1 / Hyp_ll + Hyp_p4; // expon. tail left
676  Hyp_p6 = Hyp_f5 / Hyp_lr + Hyp_p5; // expon. tail right
677  }
678 
679  while (1) {
680  // generate uniform number U -- U(0, p6)
681  // case distinction corresponding to U
682  if ((U = Random() * Hyp_p6) < Hyp_p2) { // centre left
683 
684  // immediate acceptance region R2 = [k2, mode) *[0, f2), X = k2, ... mode -1
685  if ((W = U - Hyp_p1) < 0.) return(Hyp_k2 + (int32_t)(U/Hyp_f2));
686  // immediate acceptance region R1 = [k1, k2)*[0, f1), X = k1, ... k2-1
687  if ((Y = W / Hyp_dl) < Hyp_f1) return(Hyp_k1 + (int32_t)(W/Hyp_f1));
688 
689  // computation of candidate X < k2, and its reflected counterpart V > k2
690  // either squeeze-acceptance of X or acceptance-rejection of V
691  Dk = (int32_t)(Hyp_dl * Random()) + 1;
692  if (Y <= Hyp_f2 - Dk * (Hyp_f2 - Hyp_f2/Hyp_r2)) { // quick accept of
693  return(Hyp_k2 - Dk);} // X = k2 - Dk
694 
695  if ((W = Hyp_f2 + Hyp_f2 - Y) < 1.) { // quick reject of V
696  V = Hyp_k2 + Dk;
697  if (W <= Hyp_f2 + Dk * (1. - Hyp_f2)/(Hyp_dl + 1.)) { // quick accept of V
698  return(V);
699  }
700  if (log(W) <= Hyp_c_pm - fc_lnpk(V, Hyp_L, m, n)) {
701  return(V); // final accept of V
702  }
703  }
704  X = Hyp_k2 - Dk; // go to final accept/reject
705  }
706 
707  else if (U < Hyp_p4) { // centre right
708 
709  // immediate acceptance region R3 = [mode, k4+1)*[0, f4), X = mode, ... k4
710  if ((W = U - Hyp_p3) < 0.) return(Hyp_k4 - (int32_t)((U - Hyp_p2)/Hyp_f4));
711 
712  // immediate acceptance region R4 = [k4+1, k5+1)*[0, f5)
713  if ((Y = W / Hyp_dr) < Hyp_f5) return(Hyp_k5 - (int32_t)(W/Hyp_f5));
714 
715  // computation of candidate X > k4, and its reflected counterpart V < k4
716  // either squeeze-acceptance of X or acceptance-rejection of V
717  Dk = (int32_t)(Hyp_dr * Random()) + 1;
718  if (Y <= Hyp_f4 - Dk * (Hyp_f4 - Hyp_f4*Hyp_r4)) { // quick accept of
719  return(Hyp_k4 + Dk); // X = k4 + Dk
720  }
721  if ((W = Hyp_f4 + Hyp_f4 - Y) < 1.) { // quick reject of V
722  V = Hyp_k4 - Dk;
723  if (W <= Hyp_f4 + Dk * (1. - Hyp_f4)/Hyp_dr) { // quick accept of
724  return V; // V = k4 - Dk
725  }
726 
727  if (log(W) <= Hyp_c_pm - fc_lnpk(V, Hyp_L, m, n)) {
728  return(V); // final accept of V
729  }
730  }
731  X = Hyp_k4 + Dk; // go to final accept/reject
732  }
733 
734  else {
735  Y = Random();
736  if (U < Hyp_p5) { // expon. tail left
737  Dk = (int32_t)(1. - log(Y)/Hyp_ll);
738  if ((X = Hyp_k1 - Dk) < 0) continue; // 0 <= X <= k1 - 1
739  Y *= (U - Hyp_p4) * Hyp_ll; // Y -- U(0, h(x))
740  if (Y <= Hyp_f1 - Dk * (Hyp_f1 - Hyp_f1/Hyp_r1)) {
741  return X; // quick accept of X
742  }
743  }
744  else { // expon. tail right
745  Dk = (int32_t)(1. - log(Y)/Hyp_lr);
746  if ((X = Hyp_k5 + Dk) > n ) continue; // k5 + 1 <= X <= n
747  Y *= (U - Hyp_p5) * Hyp_lr; // Y -- U(0, h(x))
748  if (Y <= Hyp_f5 - Dk * (Hyp_f5 - Hyp_f5*Hyp_r5)) {
749  return X; // quick accept of X
750  }
751  }
752  }
753 
754  // acceptance-rejection test of candidate X from the original area
755  // test, whether Y <= f(X), with Y = U*h(x) and U -- U(0, 1)
756  // log f(X) = log( mode! (m - mode)! (n - mode)! (N - m - n + mode)! )
757  // - log( X! (m - X)! (n - X)! (N - m - n + X)! )
758  if (log(Y) <= Hyp_c_pm - fc_lnpk(X, Hyp_L, m, n)) return(X);
759  }
760 }
