8 * Description:
9 * This program makes a table of residues of a certain expansion of the
10 * error function, used for the Laplace method for calculating Wallenius
11 * noncentral hypergeometric distribution.
12 *
13 * Use a compiler that supports long double precision for best results.
14 *
15 * This program generates the file ERFRES.H. The program must be re-run if
16 * the constants defined in stocc.h are changed. These constants are:
17 * ERFRES_B: begin: -log2 of lowest precision
18 * ERFRES_E: end: -log2 of highest precision
19 * ERFRES_S: step size from begin to end
20 * ERFRES_N = (ERFRES_E-ERFRES_B)/ERFRES_S+1: number of tables
21 * ERFRES_L: length of each table
22 *
23 * The following tables are generated:
24 * double NumSDev[ERFRES_N]:
25 * The number of standard deviations to include in the integral =
26 * sqrt(2) * D
27 * where D = inverse error function complement (precision).
28 *
29 * double ErfRes[ERFRES_N][ERFRES_L]:
30 * Coefficients for the expansion =
31 *
32 * (2*j-1)!! inf. D^(2*k+1)*2^(k+1)
33 * -------------- * exp(-D^2) * SUMMA -------------------
34 * (2*j)! * 2^j k=j (2*k+1)!!
35 *
36 * The first index, e, indicates the precision according to
37 * precision = 2^(-ERFRES_B - e * ERFRES_S).
38 * The second index, j, is the coefficient number going from 0 to ERFRES_L-1.
39 *
40 *
41 * Instructions:
42 * 1. This program is needed only if the parameters in stocc.h are changed.
43 * 2. Define the desired parameters in stocc.h
44 * 3. Use a compiler that supports long double.
45 * 4. Compile for console mode and run.
46 * 5. Replace the old version of the file ERFRES.CPP with ERFRES.H generated here.
47 *
48 * Further documentation:
49 * The file ran-instructions.pdf contains further documentation and
50 * instructions.
51 *
52 * Copyright 2004-2008 by Agner Fog.
53 * GNU General Public License http://www.gnu.org/licenses/gpl.html
54 *****************************************************************************/

69 double Erf (double x) {
70  // Calculates the error function erf(x) as a series expansion or
71  // continued fraction expansion
72  if (x < 0.) return -Erf(-x);
73  if (x > 10.) return 1.;
74  if (x < 2.4) {
75  // use series expansion
76  double term; // term in summation
77  double j21; // 2j+1
78  double sum = 0; // summation
79  double xx2 = x*x*2.; // 2x^2
80  int j;
81  term = x; j21 = 1.;
82  for (j=0; j<=50; j++) { // summation loop
83  sum += term;
84  if (term <= 1.E-13) break;
85  j21 += 2.;
86  term *= xx2 / j21;}
87  return exp(-x*x) * sum * rsqrtpi2;}
88  else {
89  // use continued fraction expansion
90  double a, f;
91  int n = int(2.25f*x*x - 23.4f*x + 60.84f); // predict expansion degree
92  if (n < 1) n = 1;
93  a = 0.5 * n; f = x;
94  for (; n > 0; n--) { // continued fraction loop
95  f = x + a / f;
96  a -= 0.5;
97  }
98  return 1. - exp(-x*x) * rsqrtpi / f;
99  }
100 }

103 double ErfC (double x) {
104 /* Calculates the error function complement erfc(x) as the
105  continued fraction expansion:
106 
107  erfc(x) = exp(-x^2)/sqrt(pi) * 1/(x+0.5/(x+1.0/(x+1.5/(x+ ...))))
108 
109  This continued fraction is computed forwards in order to facilitate detecting
110  when to stop the expansion.
111  The method for computing a continued fraction forwards is described in:
112  Peter Henrici: "Applied and computational complex analysis", vol. 2, p. 483, 1977.
113 */
114 
115  if (x < -1.) return 2. - ErfC(-x);
116  if (x < 1.) return 1. - Erf(x);
117 
118  double a, f, d, q1, q2, q3, m, pa, xr, xr2;
119 
120  xr = 1. / x; xr2 = xr * xr;
121  f = 1.; a = 0.5; pa = a * xr2; m = -1.;
122  q1 = 1.; q2 = 1. + pa;
123 
124  do {
125  d = m * pa / (q1 * q2);
126  m = -m;
127  a += 0.5;
128  pa *= a * xr2;
129  q3 = a * xr2 * q1 + q2;
130  q1 = q2; q2 = q3;
131  f += d;
132  }
133  while (fabs(d) > 1E-14 * fabs(f));
134 
135  return exp(-x*x) * rsqrtpi * f * xr;
136 }

140 double InvErfC (double y) {
141  // Calculates the inverse error function complement erfc (y)
142  // by Newton-Raphson iteration
143  double x, dx, y1, yd;
144  x = 1.; // initial guess
145  do {
146  y1 = ErfC(x) - y;
147  yd = -rsqrtpi2 * exp (- x * x); // derivative
148  dx = y1 / yd;
149  if (dx > 5) dx = 5;
150  x -= dx;
151  }
152  while (fabs(dx) > 1E-10*fabs(x));
153  return x;
154 }

157 long double ErfResidue (long double x, int l) {
158  // Calculates sqrt(pi)/2 * erf(x) as a series expansion for l=0.
159  // Calculates the l'th residue of the expansion for l>0
160  const MAXTERMS = 2048; // maximum number of terms
161  long double term; // term in summation
162  long double k21; // 2k+1
163  long double sum = 0.; // summation
164  int j; // index
165  if (x > 50 || x < 0) {
166  printf("parameter out of range in function ErfResidue");
167  exit(1);}
168 
169  term = 2 * x; k21 = 1.;
170  for (j = 0; j < MAXTERMS; ) {
171  if (j >= l) sum += term;
172  k21 += 2.;
173  term *= 2*x*x/k21;
174  j++;
175  if (fabs(term) <= 1.E-100) break;
176  }
177  if (term > 1.E-16) {
178  printf("Expansion too long in function ErfResidue");
179  exit(1);
180  }
181  return sum * expl(-x*x);
182 }
