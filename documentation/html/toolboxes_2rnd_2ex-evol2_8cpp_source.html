13 * discreete non-overlapping generations.
14 *
15 * The ab locus can contain genes a and b. In the initial state, all
16 * individuals have genotype (a,a).
17 * Gene b can be recessive or dominant or partially dominant.
18 *
19 * Selection takes place in the form of differential fertility, not
20 * differential survival. The selective advantage of father and mother can
21 * be added or multiplied.
22 *
23 * Instructions:
24 * Set the parameters to the desired values. The parameters are defined at
25 * "Define parameters" below. Compile for console mode. The necessary source

48 void SplitIntoGenotypes(int32_t * genotypes, int32_t * genes, StochasticLib1 & stoc) {
49  // split gene pool into (a,a), (a,b) and (b,b) genotypes
50  // parameters:
51  // genotypes: output. array with 3 places.
52  // gets number of (a,a), (a,b), and (b,b) genotypes, respectively
53  // genes: gene pool. array with 2 places
54  // stoc: random library
55 
56  int32_t g; // total number of genes
57  int32_t na_; // number of individuals with gene a on first place
58 
59  g = genes[0] + genes[1];
60  na_ = stoc.Hypergeometric(g/2, genes[0], g);
61  genotypes[0] = stoc.Hypergeometric(na_, genes[0] - na_, g/2);
62  genotypes[1] = genes[0] - 2*genotypes[0];
63  genotypes[2] = g/2 - genotypes[0] - genotypes[1];}

66 //-----------------------------------------------------------------------------
67 // function BreedingSuccess
68 //-----------------------------------------------------------------------------
69 int BreedingSuccess(double * ChildGenotypes, int32_t * FathersGenotypes,
70  int32_t * MothersGenotypes, double * Selection, int Mul) {
71  // Calculate the probability of each child genotype from the parent genotypes.
72  // Genotypes are:
73  // 0 = (a,a)
74  // 1 = (a,b)
75  // 2 = (b,b)
76  // Parameters:
77  // ChildGenotypes: Results. The relative probability of each genotype in children.
78  // FathersGenotypes: The number of male parents of each genotype
79  // MothersGenotypes: The number of female parents of each genotype
80  // Selection: The selection coefficient in favor of each genotype
81  // Mul: 0 if fitness of father and mother is added,
82  // 1 if fitness of father and mother is multiplied.
83  // return value: 1 if success, 0 if no parents of one sex
84 
85  int i, j;
86  double f;
87 
88  for (i=0; i<3; i++) ChildGenotypes[i] = 0;
89 
90  for (i=0; i<3; i++) {
91  for (j=i; j<3; j++) {
92  // compute for father with genotype i and mother with genotype j and
93  // vice versa.
94  // compute fitness of pair
95  if (Mul) {
96  f = (1. + Selection[i]) * (1. + Selection[j]);}
97  else {
98  f = 1. + Selection[i] + Selection[j];}
99  // multiply with probability of mating between i and j or j and i.
100  if (i == j) {
101  f *= (double)FathersGenotypes[i] * MothersGenotypes[i];}
102  else {
103  f *= (double)FathersGenotypes[i] * MothersGenotypes[j]
104  + (double)FathersGenotypes[j] * MothersGenotypes[i];}
105 
106  // probability of each child genotype from i,j pair
107  switch(3*i+j) {
108  case 3*0+0: // (a,a) x (a,a) -> (a,a)
109  ChildGenotypes[0] += 4*f;
110  break;
111  case 3*0+1: // (a,a) x (a,b) -> (a,a) or (a,b)
112  ChildGenotypes[0] += 2*f;
113  ChildGenotypes[1] += 2*f;
114  break;
115  case 3*0+2: // (a,a) x (b,b) -> (a,b)
116  ChildGenotypes[1] += 4*f;
117  break;
118  case 3*1+1: // (a,b) x (a,b) -> (a,a) + 2(a,b) + (b,b)
119  ChildGenotypes[0] += f;
120  ChildGenotypes[1] += 2*f;
121  ChildGenotypes[2] += f;
122  break;
123  case 3*1+2: // (a,b) x (b,b) -> (a,b) or (b,b)
124  ChildGenotypes[1] += 2*f;
125  ChildGenotypes[2] += 2*f;
126  break;
127  case 3*2+2: // (b,b) x (b,b) -> (b,b)
128  ChildGenotypes[2] += 4*f;
129  break;
130  }
131  }
132  }
133  for (i=0, f=0; i<3; i++) f += ChildGenotypes[i];
134  return f != 0;
135 }

139 // main

141 int main() {
142 
143  //---------------------------------------------------------------------------
144  // Define parameters
145  // You may change these parameters:
146  //---------------------------------------------------------------------------
147  int32_t PopulationSize = 100; // initial population size
148  double MutationRate = 1.E-5; // mutation rate
149  double GrowthRate = 1.10; // unrestrained growth rate
150  int32_t CarryingCapacity = 100000; // maximum number of individuals in habitat
151  double CapacityDeviation = 1000; // standard deviation of max individuals
152  double SelectionCoefficient = 0.1; // coefficient of selection in favor of B over A
153  double Dominance = 0.5; // 0 if gene a dominant,
154  // 1 if gene b dominant,
155  // between 0 and 1 if partial dominance
156  int Mul = 1; // 0 if fitness of father and mother is added
157  // 1 if fitness of father and mother is multiplied
158  double SexRatio = 0.5; // fraction of males
159  int32_t Generations = 4000; // number of generations to simulate
160 
161  //---------------------------------------------------------------------------
162  // other declarations
163  //---------------------------------------------------------------------------
164  // make random library
165  int32_t seed = (int32_t)time(0); // random seed
166  StochasticLib1 sto(seed); // make instance of random library
167 
168  // other variables
169  int32_t ParentGenePool[2]; // number of genes a and b
170  int32_t ChildGenePool[2]; // gene pool of child generation
171  int32_t GenerationNumber; // generation number
172  int32_t Mutations; // forward - backwards mutations
173  int32_t Children; // number of individuals in child population
174  int32_t MaxPopulation; // maximum population size
175  int32_t GenotypesFemales[3]; // genotypes (a,a), (a,b), and (b,b)
176  int32_t GenotypesMales[3]; // same, males only
177  int32_t GenotypesChildren[3]; // same, children
178  double ChildProb[3]; // probability distribution of child genotypes
179  double GenoSelection[3]; // selection coefficient of each genotype
180  int32_t OutputPeriod; // period between outputs
181  int32_t i; // loop counter
182 
183  //---------------------------------------------------------------------------
184  // initializations
185  //---------------------------------------------------------------------------
186  // compute selection coefficients
187  GenoSelection[0] = 0;
188  GenoSelection[1] = SelectionCoefficient * Dominance;
189  GenoSelection[2] = SelectionCoefficient;
190 
191  // make initial gene pool
192  ParentGenePool[0] = 2*PopulationSize;
193  ParentGenePool[1] = 0;
194 
195  // Prepare output
196  OutputPeriod = Generations / 20;
197  printf("generation gene a gene b");
198  printf("\n %8i %8li %8li", 0, ParentGenePool[0], ParentGenePool[1]);
199 
200  //---------------------------------------------------------------------------
201  // Generation loop
202  //---------------------------------------------------------------------------
203  for (GenerationNumber = 1; GenerationNumber <= Generations; GenerationNumber++) {
204 
205  // mutation
206  Mutations = sto.Binomial(ParentGenePool[0], MutationRate)
207  - sto.Binomial(ParentGenePool[1], MutationRate);
208  ParentGenePool[0] -= Mutations;
209  ParentGenePool[1] += Mutations;
210 
211  // split parent gene pool into (a,a), (a,b) and (b,b) genotypes
212  SplitIntoGenotypes(GenotypesFemales, ParentGenePool, sto);
213 
214  // split into males and females
215  for (i=0; i<3; i++) {
216  GenotypesMales[i] = sto.Binomial(GenotypesFemales[i], SexRatio);
217  GenotypesFemales[i] -= GenotypesMales[i];
218  }
219 
220  // breeding potential. Poisson or normal distribution
221  Children = sto.Poisson(PopulationSize * GrowthRate);
222 
223  // limit population size
224  PopulationSize = Children;
225  MaxPopulation = (int32_t)sto.Normal(CarryingCapacity, CapacityDeviation);
226  if (MaxPopulation < 0) MaxPopulation = 0;
227  if (PopulationSize > MaxPopulation) PopulationSize = MaxPopulation;
228 
229  // differential fertility
230  i = BreedingSuccess(ChildProb, GenotypesMales, GenotypesFemales, GenoSelection, Mul);
231  if (!i) {
232  // no offspring because all parents are of same sex
233  break;
234  }
235 
236  // Breed. Child genotypes follows multinomial distribution
237  sto.Multinomial(GenotypesChildren, ChildProb, PopulationSize, 3);
238 
239  // new gene pool
240  ChildGenePool[0] = GenotypesChildren[0]*2 + GenotypesChildren[1];
241  ChildGenePool[1] = GenotypesChildren[1] + GenotypesChildren[2]*2;
242 
243  // generation shift
244  ParentGenePool[0] = ChildGenePool[0];
245  ParentGenePool[1] = ChildGenePool[1];
246 
247  // Output
248  if (GenerationNumber % OutputPeriod == 0) {
249  printf("\n %8li %8li %8li",
250  GenerationNumber, ParentGenePool[0], ParentGenePool[1]);
251  if (PopulationSize == 0) break;
252  }
253 
254  //-------------------------------------------------------------------------
255  // end of generation loop
256  //-------------------------------------------------------------------------
257  }
258 
259  EndOfProgram(); // system-specific exit code
260  return 0;
261 }
