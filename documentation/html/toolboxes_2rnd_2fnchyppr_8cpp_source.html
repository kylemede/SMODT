54 int32_t CFishersNCHypergeometric::mode(void) {
55  // Find mode (exact)
56  // Uses the method of Liao and Rosen, The American Statistician, vol 55,
57  // no 4, 2001, p. 366-369.
58  // Note that there is an error in Liao and Rosen's formula.
59  // Replace sgn(b) with -1 in Liao and Rosen's formula.
60 
61  double A, B, C, D; // coefficients for quadratic equation
62  double x; // mode
63  int32_t L = m + n - N;
64  int32_t m1 = m+1, n1 = n+1;
65 
66  if (odds == 1.) {
67  // simple hypergeometric
68  x = (m + 1.) * (n + 1.) / (N + 2.);
69  }
70  else {
71  // calculate analogously to Cornfield mean
72  A = 1. - odds;
73  B = (m1+n1)*odds - L;
74  C = -(double)m1*n1*odds;
75  D = B*B -4*A*C;
76  D = D > 0. ? sqrt(D) : 0.;
77  x = (D - B)/(A+A);
78  }
79  return (int32_t)x;
80 }

83 double CFishersNCHypergeometric::mean(void) {
84  // Find approximate mean
85  // Calculation analogous with mode
86  double a, b; // temporaries in calculation
87  double mean; // mean
88 
89  if (odds == 1.) { // simple hypergeometric
90  return double(m)*n/N;
91  }
92  // calculate Cornfield mean
93  a = (m+n)*odds + (N-m-n);
94  b = a*a - 4.*odds*(odds-1.)*m*n;
95  b = b > 0. ? sqrt(b) : 0.;
96  mean = (a-b)/(2.*(odds-1.));
97  return mean;
98 }

101 double CFishersNCHypergeometric::variance(void) {
102  // find approximate variance (poor approximation)
103  double my = mean(); // approximate mean
104  // find approximate variance from Fisher's noncentral hypergeometric approximation
105  double r1 = my * (m-my); double r2 = (n-my)*(my+N-n-m);
106  double var = N*r1*r2/((N-1)*(m*r2+(N-m)*r1));
107  if (var < 0.) var = 0.;
108  return var;

111 double CFishersNCHypergeometric::moments(double * mean_, double * var_) {
112  // calculate exact mean and variance
113  // return value = sum of f(x), expected = 1.
114  double y, sy=0, sxy=0, sxxy=0, me1;
115  int32_t x, xm, x1;
116  const double accur = 0.1 * accuracy; // accuracy of calculation
117  xm = (int32_t)mean(); // approximation to mean
118  for (x=xm; x<=xmax; x++) {
119  y = probability(x);
120  x1 = x - xm; // subtract approximate mean to avoid loss of precision in sums
121  sy += y; sxy += x1 * y; sxxy += x1 * x1 * y;
122  if (y < accur) break;
123  }
124  for (x=xm-1; x>=xmin; x--) {
125  y = probability(x);
126  x1 = x - xm; // subtract approximate mean to avoid loss of precision in sums
127  sy += y; sxy += x1 * y; sxxy += x1 * x1 * y;
128  if (y < accur) break;
129  }
130  me1 = sxy / sy;
131  *mean_ = me1 + xm;
132  y = sxxy / sy - me1 * me1;

139 double CFishersNCHypergeometric::probability(int32_t x) {
140  // calculate probability function
141  const double accur = accuracy * 0.1;// accuracy of calculation
142 
143  if (x < xmin || x > xmax) return 0;
144  if (!rsum) {
145  // first time. calculate rsum = reciprocal of sum of proportional
146  // function over all probable x values
147  int32_t x1, x2; // x loop
148  double y; // value of proportional function
149  x1 = (int32_t)mean(); // start at mean
150  if (x1 < xmin) x1 = xmin;
151  x2 = x1 + 1;
152  scale = 0.; scale = lng(x1); // calculate scale to avoid overflow
153  rsum = 1.; // = exp(lng(x1)) with this scale
154  for (x1--; x1 >= xmin; x1--) {
155  rsum += y = exp(lng(x1)); // sum from x1 and down
156  if (y < accur) break; // until value becomes negligible
157  }
158  for (; x2 <= xmax; x2++) { // sum from x2 and up
159  rsum += y = exp(lng(x2));
160  if (y < accur) break; // until value becomes negligible
161  }
162  rsum = 1. / rsum; // save reciprocal sum
163  }
164  return exp(lng(x)) * rsum; // function value
165 }

167 double CFishersNCHypergeometric::probabilityRatio(int32_t x, int32_t x0) {
168  // Calculate probability ratio f(x)/f(x0)
169  // This is much faster than calculating a single probability because
170  // rsum is not needed
171  double a1, a2, a3, a4, f1, f2, f3, f4;
172  int32_t y, dx = x - x0;
173  int invert = 0;
174 
175  if (x < xmin || x > xmax) return 0.;
176  if (x0 < xmin || x0 > xmax) {
177  FatalError("Infinity in CFishersNCHypergeometric::probabilityRatio");
178  }
179  if (dx == 0.) return 1.;
180  if (dx < 0.) {
181  invert = 1;
182  dx = -dx;
183  y = x; x = x0; x0 = y;
184  }
185  a1 = m - x0; a2 = n - x0; a3 = x; a4 = N - m - n + x;
186  if (dx <= 28 && x <= 100000) { // avoid overflow
187  // direct calculation
188  f1 = f2 = 1.;
189  // compute ratio of binomials
190  for (y = 0; y < dx; y++) {
191  f1 *= a1-- * a2--;
192  f2 *= a3-- * a4--;
193  }
194  // compute odds^dx
195  f3 = 1.; f4 = odds; y = dx;
196  while (y) {
197  if (f4 < 1.E-100) {
198  f3 = 0.; break; // avoid underflow
199  }
200  if (y & 1) f3 *= f4;
201  f4 *= f4;
202  y = (unsigned long)(y) >> 1;
203  }
204  f1 = f3 * f1 / f2;
205  if (invert) f1 = 1. / f1;
206  }
207  else {
208  // use logarithms
209  f1 = FallingFactorial(a1,dx) + FallingFactorial(a2,dx) -
210  FallingFactorial(a3,dx) - FallingFactorial(a4,dx) +
211  dx * log(odds);
212  if (invert) f1 = -f1;
213  f1 = exp(f1);
214  }
215  return f1;
216 }

218 double CFishersNCHypergeometric::MakeTable(double * table, int32_t MaxLength, int32_t * xfirst, int32_t * xlast, double cutoff) {
219  // Makes a table of Fisher's noncentral hypergeometric probabilities.
220  // Results are returned in the array table of size MaxLength.
221  // The values are scaled so that the highest value is 1. The return value
222  // is the sum, s, of all the values in the table. The normalized
223  // probabilities are obtained by multiplying all values in the table by
224  // 1/s.
225  // The tails are cut off where the values are < cutoff, so that
226  // *xfirst may be > xmin and *xlast may be < xmax.
227  // The value of cutoff will be 0.01 * accuracy if not specified.
228  // The first and last x value represented in the table are returned in
229  // *xfirst and *xlast. The resulting probability values are returned in the
230  // first (*xlast - *xfirst + 1) positions of table. If this would require
231  // more than MaxLength values then the table is filled with as many
232  // correct values as possible.
233  //
234  // The function will return the desired length of table when MaxLength = 0.
235 
236  double f; // probability function value
237  double sum; // sum of table values
238  double a1, a2, b1, b2; // factors in recursive calculation of f(x)
239  int32_t x; // x value
240  int32_t x1, x2; // lowest and highest x
241  int32_t i, i0, i1, i2; // table index
242  int32_t mode = this->mode(); // mode
243  int32_t L = n + m - N; // parameter
244  int32_t DesiredLength; // desired length of table
245 
246  // limits for x
247  x1 = (L > 0) ? L : 0; // xmin
248  x2 = (n < m) ? n : m; // xmax
249 
250  if (MaxLength <= 0) {
251  // Return UseTable and LengthNeeded
252  DesiredLength = x2 - x1 + 1; // max length of table
253  if (DesiredLength > 200) {
254  double sd = sqrt(variance()); // calculate approximate standard deviation
255  // estimate number of standard deviations to include from normal distribution
256  i = (int32_t)(NumSD(accuracy) * sd + 0.5);
257  if (DesiredLength > i) DesiredLength = i;
258  }
259  if (xfirst) *xfirst = 1; // for analogy with CWalleniusNCHypergeometric::MakeTable
260  return DesiredLength;
261  }
262 
263  // place mode in the table
264  if (mode - x1 <= MaxLength/2) {
265  // There is enough space for left tail
266  i0 = mode - x1;
267  }
268  else if (x2 - mode <= MaxLength/2) {
269  // There is enough space for right tail
270  i0 = MaxLength - x2 + mode - 1;
271  if (i0 < 0) i0 = 0;
272  }
273  else {
274  // There is not enough space for any of the tails. Place mode in middle of table
275  i0 = MaxLength/2;
276  }
277  // Table start index
278  i1 = i0 - mode + x1; if (i1 < 0) i1 = 0;
279 
280  // Table end index
281  i2 = i0 + x2 - mode; if (i2 > MaxLength-1) i2 = MaxLength-1;
282 
283  // make center
284  table[i0] = sum = f = 1.;
285 
286  // make left tail
287  x = mode;
288  a1 = m + 1 - x; a2 = n + 1 - x;
289  b1 = x; b2 = x - L;
290  for (i = i0 - 1; i >= i1; i--) {
291  f *= b1 * b2 / (a1 * a2 * odds); // recursive formula
292  a1++; a2++; b1--; b2--;
293  sum += table[i] = f;
294  if (f < cutoff) {
295  i1 = i; break; // cut off tail if < accuracy
296  }
297  }
298  if (i1 > 0) {
299  // move table down for cut-off left tail
300  memcpy(table, table+i1, (i0-i1+1)*sizeof(*table));
301  // adjust indices
302  i0 -= i1; i2 -= i1; i1 = 0;
303  }
304  // make right tail
305  x = mode + 1;
306  a1 = m + 1 - x; a2 = n + 1 - x;
307  b1 = x; b2 = x - L;
308  f = 1.;
309  for (i = i0 + 1; i <= i2; i++) {
310  f *= a1 * a2 * odds / (b1 * b2); // recursive formula
311  a1--; a2--; b1++; b2++;
312  sum += table[i] = f;
313  if (f < cutoff) {
314  i2 = i; break; // cut off tail if < accuracy
315  }
316  }
317  // x limits
318  *xfirst = mode - (i0 - i1);
319  *xlast = mode + (i2 - i0);
320 
321  return sum;
322 }

325 double CFishersNCHypergeometric::lng(int32_t x) {
326  // natural log of proportional function
327  // returns lambda = log(m!*x!/(m-x)!*m2!*x2!/(m2-x2)!*odds^x)
328  int32_t x2 = n - x, m2 = N - m;
329  if (ParametersChanged) {
330  mFac = LnFac(m) + LnFac(m2);
331  xLast = -99; ParametersChanged = 0;
332  }
333  if (m < FAK_LEN && m2 < FAK_LEN) goto DEFLT;
334  switch (x - xLast) {
335  case 0: // x unchanged
336  break;
337  case 1: // x incremented. calculate from previous value
338  xFac += log (double(x) * (m2-x2) / (double(x2+1)*(m-x+1)));
339  break;
340  case -1: // x decremented. calculate from previous value
341  xFac += log (double(x2) * (m-x) / (double(x+1)*(m2-x2+1)));
342  break;
343  default: DEFLT: // calculate all
344  xFac = LnFac(x) + LnFac(x2) + LnFac(m-x) + LnFac(m2-x2);
345  }
346  xLast = x;
347  return mFac - xFac + x * logodds - scale;
348 }

380 void CMultiFishersNCHypergeometric::mean(double * mu) {
381  // calculates approximate mean of multivariate Fisher's noncentral
382  // hypergeometric distribution. Result is returned in mu[0..colors-1].
383  // The calculation is reasonably fast.
384  double r, r1; // iteration variable
385  double q; // mean of color i
386  double W; // total weight
387  int i; // color index
388  int iter = 0; // iteration counter
389 
390  if (colors < 3) {
391  // simple cases
392  if (colors == 1) mu[0] = n;
393  if (colors == 2) {
394  mu[0] = CFishersNCHypergeometric(n,m[0],m[0]+m[1],odds[0]/odds[1]).mean();
395  mu[1] = n - mu[0];
396  }
397  return;
398  }
399  if (n == N) {
400  // Taking all balls
401  for (i = 0; i < colors; i++) mu[i] = m[i];
402  return;
403  }
404 
405  // initial guess for r
406  for (i=0, W=0.; i < colors; i++) W += m[i] * odds[i];
407  r = (double)n * N / ((N-n)*W);
408 
409  // iteration loop to find r
410  do {
411  r1 = r;
412  for (i=0, q=0.; i < colors; i++) {
413  q += m[i] * r * odds[i] / (r * odds[i] + 1.);
414  }
415  r *= n * (N-q) / (q * (N-n));
416  if (++iter > 100) FatalError("convergence problem in function CMultiFishersNCHypergeometric::mean");
417  }
418  while (fabs(r-r1) > 1E-5);
419 
420  // store result
421  for (i=0; i < colors; i++) {
422  mu[i] = m[i] * r * odds[i] / (r * odds[i] + 1.);
423  }
424 }

429  // hypergeometric distribution (accuracy is not too good).

498 void CMultiFishersNCHypergeometric::SumOfAll() {
499  // this function does the very time consuming job of calculating the sum
500  // of the proportional function g(x) over all possible combinations of
501  // the x[i] values with probability > accuracy. These combinations are
502  // generated by the recursive function loop().
503  // The mean and variance are generated as by-products.
504 
505  int i; // color index
506  int32_t msum; // sum of m[i]
507 
508  // get approximate mean
509  mean(sx);
510  // round mean to integers
511  for (i=0, msum=0; i < colors; i++) {
512  msum += xm[i] = (int32_t)(sx[i]+0.4999999);}
513  // adjust truncated x values to make the sum = n
514  msum -= n;
515  for (i = 0; msum < 0; i++) {
516  if (xm[i] < m[i]) {
517  xm[i]++; msum++;
518  }
519  }
520  for (i = 0; msum > 0; i++) {
521  if (xm[i] > 0) {
522  xm[i]--; msum--;
523  }
524  }
525 
526  // adjust scale factor to g(mean) to avoid overflow
527  scale = 0.; scale = lng(xm);
528 
529  // initialize for recursive loops
530  sn = 0;
531  for (i = colors-1, msum = 0; i >= 0; i--) {
532  remaining[i] = msum; msum += m[i];
533  }
534  for (i = 0; i < colors; i++) {
535  sx[i] = 0; sxx[i] = 0;
536  }
537 
538  // recursive loops to calculate sums of g(x) over all x combinations
539  rsum = 1. / loop(n, 0);
540 
541  // calculate mean and variance
542  for (i = 0; i < colors; i++) {
543  sxx[i] = sxx[i]*rsum - sx[i]*sx[i]*rsum*rsum;
544  sx[i] = sx[i]*rsum;
545  }
546 }

551  // used by SumOfAll
552  int32_t x, x0; // x of color c
553  int32_t xmin, xmax; // min and max of x[c]
554  double s1, s2, sum = 0.; // sum of g(x) values

556

567  if (s1 < accuracy && s1 < s2) break; // stop when values become negligible
568  s2 = s1;
569  }
