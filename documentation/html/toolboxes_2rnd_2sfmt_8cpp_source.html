125  else {

146  const uint32_t factor = 1812433253U;// Multiplication factor
147  r = (uint32_t)NumSeeds;
148  for (i = 0; i < SFMT_N*4; i++) {
149  r = factor * (r ^ (r >> 30)) + i;
150  sta[i] = r;
151  }
152 
153 #endif
154 
155  // 2. loop: Fill state vector with random numbers from seeds[]
156  for (i = 1, j = 0; j < count; j++) {
157  r = func1(sta[i] ^ sta[(i + mid) % size] ^ sta[(i + size - 1) % size]);
158  sta[(i + mid) % size] += r;
159  if (j < (uint32_t)NumSeeds) r += (uint32_t)seeds[j];
160  r += i;
161  sta[(i + mid + lag) % size] += r;
162  sta[i] = r;
163  i = (i + 1) % size;
164  }
165 
166  // 3. loop: Randomize some more
167  for (j = 0; j < size; j++) {
168  r = func2(sta[i] + sta[(i + mid) % size] + sta[(i + size - 1) % size]);
169  sta[(i + mid) % size] ^= r;
170  r -= i;
171  sta[(i + mid + lag) % size] ^= r;
172  sta[i] = r;
173  i = (i + 1) % size;
174  }
175  if (UseMother) {
176  // 4. loop: Initialize MotherState
177  for (j = 0; j < 5; j++) {
178  r = func2(r) + j;
179  MotherState[j] = r + sta[2*j];
180  }
181  }
182 
183  // Further initialization and period certification
184  Init2();
185 }

188 void CRandomSFMT::Init2() {
189  // Various initializations and period certification
190  uint32_t i, j, temp;
191 
192  // Initialize mask
193  static const uint32_t maskinit[4] = {SFMT_MASK};
194  mask = _mm_loadu_si128((__m128i*)maskinit);
195 
196  // Period certification
197  // Define period certification vector
198  static const uint32_t parityvec[4] = {SFMT_PARITY};
199 
200  // Check if parityvec & state[0] has odd parity
201  temp = 0;
202  for (i = 0; i < 4; i++) {
203  temp ^= parityvec[i] & ((uint32_t*)state)[i];
204  }
205  for (i = 16; i > 0; i >>= 1) temp ^= temp >> i;
206  if (!(temp & 1)) {
207  // parity is even. Certification failed
208  // Find a nonzero bit in period certification vector
209  for (i = 0; i < 4; i++) {
210  if (parityvec[i]) {
211  for (j = 1; j; j <<= 1) {
212  if (parityvec[i] & j) {
213  // Flip the corresponding bit in state[0] to change parity
214  ((uint32_t*)state)[i] ^= j;
215  // Done. Exit i and j loops
216  i = 5; break;
217  }
218  }
219  }
220  }
221  }
222  // Generate first random numbers and set ix = 0
223  Generate();
224 }

227 // Subfunction for the sfmt algorithm
228 static inline __m128i sfmt_recursion(__m128i const &a, __m128i const &b,
229 __m128i const &c, __m128i const &d, __m128i const &mask) {
230  __m128i a1, b1, c1, d1, z1, z2;
231  b1 = _mm_srli_epi32(b, SFMT_SR1);
232  a1 = _mm_slli_si128(a, SFMT_SL2);
233  c1 = _mm_srli_si128(c, SFMT_SR2);
234  d1 = _mm_slli_epi32(d, SFMT_SL1);
235  b1 = _mm_and_si128(b1, mask);
236  z1 = _mm_xor_si128(a, a1);
237  z2 = _mm_xor_si128(b1, d1);
238  z1 = _mm_xor_si128(z1, c1);
239  z2 = _mm_xor_si128(z1, z2);
240  return z2;
241 }

243 void CRandomSFMT::Generate() {
244  // Fill state array with new random numbers
245  int i;
246  __m128i r, r1, r2;
247 
248  r1 = state[SFMT_N - 2];
249  r2 = state[SFMT_N - 1];
250  for (i = 0; i < SFMT_N - SFMT_M; i++) {
251  r = sfmt_recursion(state[i], state[i + SFMT_M], r1, r2, mask);
252  state[i] = r;
253  r1 = r2;
254  r2 = r;
255  }
256  for (; i < SFMT_N; i++) {
257  r = sfmt_recursion(state[i], state[i + SFMT_M - SFMT_N], r1, r2, mask);
258  state[i] = r;
259  r1 = r2;
260  r2 = r;
261  }
262  ix = 0;
263 }

277 uint32_t CRandomSFMT::MotherBits() {
278  // Get random bits from Mother-Of-All generator
279  uint64_t sum;
280  sum =
281  (uint64_t)2111111111U * (uint64_t)MotherState[3] +
282  (uint64_t)1492 * (uint64_t)MotherState[2] +
283  (uint64_t)1776 * (uint64_t)MotherState[1] +
284  (uint64_t)5115 * (uint64_t)MotherState[0] +
285  (uint64_t)MotherState[4];
286  MotherState[3] = MotherState[2];
287  MotherState[2] = MotherState[1];
288  MotherState[1] = MotherState[0];
289  MotherState[4] = (uint32_t)(sum >> 32); // Carry
290  MotherState[0] = (uint32_t)sum; // Low 32 bits of sum
291  return MotherState[0];
292 }

294 int CRandomSFMT::IRandom (int min, int max) {
295  // Output random integer in the interval min <= x <= max
296  // Slightly inaccurate if (max-min+1) is not a power of 2
297  if (max <= min) {
298  if (max == min) return min; else return 0x80000000;
299  }
300  // Assume 64 bit integers supported. Use multiply and shift method
301  uint32_t interval; // Length of interval
302  uint64_t longran; // Random bits * interval
303  uint32_t iran; // Longran / 2^32
304 
305  interval = (uint32_t)(max - min + 1);
306  longran = (uint64_t)BRandom() * interval;
307  iran = (uint32_t)(longran >> 32);
308  // Convert back to signed and return result
309  return (int32_t)iran + min;
310 }

312 int CRandomSFMT::IRandomX (int min, int max) {
313  // Output random integer in the interval min <= x <= max
314  // Each output value has exactly the same probability.
315  // This is obtained by rejecting certain bit values so that the number
316  // of possible bit values is divisible by the interval length
317  if (max <= min) {
318  if (max == min) {
319  return min; // max == min. Only one possible value
320  }
321  else {
322  return 0x80000000; // max < min. Error output
323  }
324  }
325  // Assume 64 bit integers supported. Use multiply and shift method
326  uint32_t interval; // Length of interval
327  uint64_t longran; // Random bits * interval
328  uint32_t iran; // Longran / 2^32
329  uint32_t remainder; // Longran % 2^32
330 
331  interval = (uint32_t)(max - min + 1);
332  if (interval != LastInterval) {
333  // Interval length has changed. Must calculate rejection limit
334  // Reject when remainder = 2^32 / interval * interval
335  // RLimit will be 0 if interval is a power of 2. No rejection then.
336  RLimit = (uint32_t)(((uint64_t)1 << 32) / interval) * interval - 1;
337  LastInterval = interval;
338  }
339  do { // Rejection loop
340  longran = (uint64_t)BRandom() * interval;
341  iran = (uint32_t)(longran >> 32);
342  remainder = (uint32_t)longran;
343  } while (remainder > RLimit);
344  // Convert back to signed and return result
345  return (int32_t)iran + min;
346 }

348 double CRandomSFMT::Random() {
349  // Output random floating point number
350  if (ix >= SFMT_N*4-1) {
351  // Make sure we have at least two 32-bit numbers
352  Generate();
353  }
354  uint64_t r = *(uint64_t*)((uint32_t*)state+ix);
355  ix += 2;
356  if (UseMother) {
357  // We need 53 bits from Mother-Of-All generator
358  // Use the regular 32 bits and the the carry bits rotated
359  uint64_t r2 = (uint64_t)MotherBits() << 32;
360  r2 |= (MotherState[4] << 16) | (MotherState[4] >> 16);
361  r += r2;
362  }
363  // 53 bits resolution:
364  // return (int64_t)(r >> 11) * (1./(67108864.0*134217728.0)); // (r >> 11)*2^(-53)
365  // 52 bits resolution for compatibility with assembly version:
366  return (int64_t)(r >> 12) * (1./(67108864.0*67108864.0)); // (r >> 12)*2^(-52)
367 }
