70 int32_t StochasticLib3::WalleniusNCHyp (int32_t n, int32_t m, int32_t N, double odds) {
71  /*
72  This function generates a random variate with Wallenius noncentral
73  hypergeometric distribution.
74 
75  Wallenius noncentral hypergeometric distribution is the distribution you
76  get when drawing balls without replacement from an urn containing red and
77  white balls, with bias.
78 
79  We define the weight of the balls so that the probability of taking a
80  particular ball is proportional to its weight. The value of odds is the
81  normalized odds ratio: odds = weight(red) / weight(white).
82  If all balls have the same weight, i.e. odds = 1, then we get the
83  hypergeometric distribution.
84 
85  n is the number of balls you take,
86  m is the number of red balls in the urn,
87  N is the total number of balls in the urn,
88  odds is the odds ratio,
89  and the return value is the number of red balls you get.
90 
91  Four different calculation methods are implemented. This function decides
92  which method to use, based on the parameters.
93  */
94 
95  // check parameters
96  if (n >= N || m >= N || n <= 0 || m <= 0 || odds <= 0.) {
97  // trivial cases
98  if (n == 0 || m == 0) return 0;
99  if (m == N) return n;
100  if (n == N) return m;
101  if (odds == 0.) {
102  if (n > N-m) FatalError("Not enough items with nonzero weight in function WalleniusNCHyp");
103  return 0;}
104  // illegal parameter
105  FatalError("Parameter out of range in function WalleniusNCHyp");}
106 
107  if (odds == 1.) {
108  // use hypergeometric function if odds == 1
109  return Hypergeometric(n, m, N);}
110 
111  if (n < 30) {
112  return WalleniusNCHypUrn(n, m, N, odds);}
113 
114  if (double(n)*N < 10000) {
115  return WalleniusNCHypTable(n, m, N, odds);}
116 
117  return WalleniusNCHypRatioOfUnifoms(n, m, N, odds);
118  // the decision to use NoncentralHypergeometricInversion is
119  // taken inside WalleniusNCHypRatioOfUnifoms based
120  // on the calculated variance.
121 }

128 int32_t StochasticLib3::WalleniusNCHypUrn (int32_t n, int32_t m, int32_t N, double odds) {
129  // sampling from Wallenius noncentral hypergeometric distribution
130  // by simulating urn model
131  int32_t x; // sample
132  int32_t m2; // items of color 2 in urn
133  double mw1, mw2; // total weight of balls of color 1 or 2
134  x = 0; m2 = N - m;
135  mw1 = m * odds; mw2 = m2;
136  do {
137  if (Random() * (mw1 + mw2) < mw1) {
138  x++; m--;
139  if (m == 0) break;
140  mw1 = m * odds;
141  }
142  else {
143  m2--;
144  if (m2 == 0) {
145  x += n-1; break;
146  }
147  mw2 = m2;
148  }
149  }
150  while (--n);
151  return x;
152 }

155 int32_t StochasticLib3::WalleniusNCHypTable (int32_t n, int32_t m, int32_t N, double odds) {
156  // Sampling from Wallenius noncentral hypergeometric distribution
157  // using chop-down search from a table created by recursive calculation.
158  // This method is fast when n is low or when called repeatedly with
159  // the same parameters.
160 
161  int32_t x2; // upper x limit for table
162  int32_t x; // sample
163  double u; // uniform random number
164  int success; // table long enough
165 
166  if (n != wnc_n_last || m != wnc_m_last || N != wnc_N_last || odds != wnc_o_last) {
167  // set-up: This is done only when parameters have changed
168  wnc_n_last = n; wnc_m_last = m; wnc_N_last = N; wnc_o_last = odds;
169 
170  CWalleniusNCHypergeometric wnch(n,m,N,odds); // make object for calculation
171  success = wnch.MakeTable(wall_ytable, WALL_TABLELENGTH, &wall_x1, &x2); // make table of probability values
172  if (success) {
173  wall_tablen = x2 - wall_x1 + 1; // table long enough. remember length
174  }
175  else {
176  wall_tablen = 0; // remember failure
177  }
178  }
179 
180  if (wall_tablen == 0) {
181  // table not long enough. Use another method
182  return WalleniusNCHypRatioOfUnifoms(n,m,N,odds);
183  }
184 
185  while (1) { // repeat in the rare case of failure
186  u = Random(); // uniform variate to convert
187  for (x=0; x<wall_tablen; x++) { // chop-down search
188  u -= wall_ytable[x];
189  if (u < 0.) return x + wall_x1; // value found
190  }
191  }
192 }

195 int32_t StochasticLib3::WalleniusNCHypRatioOfUnifoms (int32_t n, int32_t m, int32_t N, double odds) {
196  // sampling from Wallenius noncentral hypergeometric distribution
197  // using ratio-of-uniforms rejection method.
198  int32_t xmin, xmax; // x limits
199  double mean; // mean
200  double variance; // variance
201  double x; // real sample
202  int32_t xi; // integer sample
203  int32_t x2; // limit when searching for mode
204  double u; // uniform random
205  double f, f2; // probability function value
206  double s123; // components 1,2,3 of hat width
207  double s4; // component 4 of hat width
208  double r1, r2; // temporaries
209  static const double rsqrt2pi = 0.3989422804014326857; // 1/sqrt(2*pi)
210 
211  // Make object for calculating mean and probability.
212  CWalleniusNCHypergeometric wnch(n, m, N, odds, accuracy);
213 
214  xmin = m+n-N; if (xmin < 0) xmin = 0; // calculate limits
215  xmax = n; if (xmax > m) xmax = m;
216 
217  if (n != wnc_n_last || m != wnc_m_last || N != wnc_N_last || odds != wnc_o_last) {
218  // set-up: This is done only when parameters have changed
219  wnc_n_last = n; wnc_m_last = m; wnc_N_last = N; wnc_o_last = odds;
220 
221  // find approximate mean
222  mean = wnch.mean();
223 
224  // find approximate variance from Fisher's noncentral hypergeometric approximation
225  r1 = mean * (m-mean); r2 = (n-mean)*(mean+N-n-m);
226  variance = N*r1*r2/((N-1)*(m*r2+(N-m)*r1));
227  UseChopDown = variance < 4.; // use chop-down method if variance is low
228 
229  if (!UseChopDown) {
230  // find mode (same code in CWalleniusNCHypergeometric::mode)
231  wnc_mode = (int32_t)(mean); f2 = 0.;
232  if (odds < 1.) {
233  if (wnc_mode < xmax) wnc_mode++;
234  x2 = xmin;
235  if (odds > 0.294 && N <= 10000000) {
236  x2 = wnc_mode - 1;} // search for mode can be limited
237  for (xi = wnc_mode; xi >= x2; xi--) {
238  f = wnch.probability(xi);
239  if (f <= f2) break;
240  wnc_mode = xi; f2 = f;
241  }
242  }
243  else {
244  if (wnc_mode < xmin) wnc_mode++;
245  x2 = xmax;
246  if (odds < 3.4 && N <= 10000000) {
247  x2 = wnc_mode + 1;} // search for mode can be limited
248  for (xi = wnc_mode; xi <= x2; xi++) {
249  f = wnch.probability(xi);
250  if (f <= f2) break;
251  wnc_mode = xi; f2 = f;
252  }
253  }
254  wnc_k = f2; // value at mode
255 
256  // find approximate variance from normal distribution approximation
257  variance = rsqrt2pi / wnc_k; variance *= variance;
258 
259  // find center and width of hat function
260  wnc_a = mean + 0.5;
261  s123 = 0.40 + 0.8579*sqrt(variance+0.5) + 0.4*fabs(mean-wnc_mode);
262  s4 = 0.;
263  r1 = xmax - mean - s123; r2 = mean - s123 - xmin;
264  if (r1 > r2) r1 = r2;
265  if ((odds>5. || odds<0.2) && r1>=-0.5 && r1<=8.) {
266  // s4 correction needed
267  if (r1 < 1.) r1 = 1.;
268  s4 = 0.029 * pow(double(N),0.23) / (r1*r1);
269  }
270  wnc_h = 2. * (s123 + s4);
271 
272  // find safety bounds
273  wnc_bound1 = (int32_t)(mean - 4. * wnc_h);
274  if (wnc_bound1 < xmin) wnc_bound1 = xmin;
275  wnc_bound2 = (int32_t)(mean + 4. * wnc_h);
276  if (wnc_bound2 > xmax) wnc_bound2 = xmax;
277  }
278  }
279 
280  if (UseChopDown) { // for small variance, use chop down inversion
281  return WalleniusNCHypInversion(n,m,N,odds);
282  }
283 
284  // use ratio-of-uniforms rejection method
285  while(1) { // rejection loop
286  u = Random();
287  if (u == 0.) continue; // avoid division by 0
288  x = wnc_a + wnc_h * (Random()-0.5)/u;
289  if (x < 0. || x > 2E9) continue; // reject, avoid overflow
290  xi = (int32_t)(x); // truncate
291  if (xi < wnc_bound1 || xi > wnc_bound2) {
292  continue; // reject if outside safety bounds
293  }
294 #if 0 // use rejection in x-domain
295  if (xi == wnc_mode) break; // accept
296  f = wnch.probability(xi); // function value
297  if (f > wnc_k * u * u) {
298  break; // acceptance
299  }
300 #else // use rejection in t-domain (this is faster)
301  double hx, s2, xma2; // compute h(x)
302  s2 = wnc_h * 0.5; s2 *= s2;
303  xma2 = xi - (wnc_a-0.5);
304  xma2 *= xma2;
305  hx = (s2 >= xma2) ? 1. : s2 / xma2;
306  // rejection in t-domain implemented in CWalleniusNCHypergeometric::BernouilliH
307  if (wnch.BernouilliH(xi, hx * wnc_k * 1.01, u * u * wnc_k * 1.01, this)) {
308  break; // acceptance
309  }
310 #endif
311  } // rejection
312  return xi;
313 }

316 int32_t StochasticLib3::WalleniusNCHypInversion (int32_t n, int32_t m, int32_t N, double odds) {
317  // sampling from Wallenius noncentral hypergeometric distribution
318  // using down-up search starting at the mean using the chop-down technique.
319  // This method is faster than the rejection method when the variance is low.
320  int32_t wall_x1, x2; // search values
321  int32_t xmin, xmax; // x limits
322  double u; // uniform random number to be converted
323  double f; // probability function value
324  double accura; // absolute accuracy
325  int updown; // 1 = search down, 2 = search up, 3 = both
326 
327  // Make objects for calculating mean and probability.
328  // It is more efficient to have two identical objects, one for down search
329  // and one for up search, because they are obtimized for consecutive x values.
330  CWalleniusNCHypergeometric wnch1(n, m, N, odds, accuracy);
331  CWalleniusNCHypergeometric wnch2(n, m, N, odds, accuracy);
332 
333  accura = accuracy * 0.01;
334  if (accura > 1E-7) accura = 1E-7; // absolute accuracy
335 
336  wall_x1 = (int32_t)(wnch1.mean()); // start at floor and ceiling of mean
337  x2 = wall_x1 + 1;
338  xmin = m+n-N; if (xmin<0) xmin = 0; // calculate limits
339  xmax = n; if (xmax>m) xmax = m;
340  updown = 3; // start searching both up and down
341 
342  while(1) { // loop until accepted (normally executes only once)
343  u = Random(); // uniform random number to be converted
344  while (updown) { // search loop
345  if (updown & 1) { // search down
346  if (wall_x1 < xmin) {
347  updown &= ~1;} // stop searching down
348  else {
349  f = wnch1.probability(wall_x1);
350  u -= f; // subtract probability until 0
351  if (u <= 0.) return wall_x1;
352  wall_x1--;
353  if (f < accura) updown &= ~1; // stop searching down
354  }
355  }
356  if (updown & 2) { // search up
357  if (x2 > xmax) {
358  updown &= ~2; // stop searching up
359  }
360  else {
361  f = wnch2.probability(x2);
362  u -= f; // subtract probability until 0
363  if (u <= 0.) return x2;
364  x2++;
365  if (f < accura) updown &= ~2; // stop searching down
366  }
367  }
368  }
369  }
370 }

377 void StochasticLib3::MultiWalleniusNCHyp (int32_t * destination,
378 int32_t * source, double * weights, int32_t n, int colors) {
379  /*
380  This function generates a vector of random variables with the
381  multivariate Wallenius noncentral hypergeometric distribution.
382 
383  The multivariate Wallenius noncentral hypergeometric distribution is
384  the distribution you get when drawing colored balls from an urn
385  with any number of colors, without replacement, and with bias.
386 
387  The weights are defined so that the probability of taking a particular
388  ball is proportional to its weight.
389 
390  Parameters:
391  destination: An output array to receive the number of balls of each
392  color. Must have space for at least 'colors' elements.
393  source: An input array containing the number of balls of each
394  color in the urn. Must have 'colors' elements.
395  All elements must be non-negative.
396  weights: The odds of each color. Must have 'colors' elements.
397  All elements must be non-negative.
398  n: The number of balls to draw from the urn.
399  Cannot exceed the total number of balls with nonzero weight
400  in source.
401  colors: The number of possible colors.
402 
403  MAXCOLORS (defined in stocc.h): You may adjust MAXCOLORS to the maximum
404  number of colors you need.
405 
406  The function will reduce the number of colors, if possible, by eliminating
407  colors with zero weight or zero number and pooling together colors with the
408  same weight. The problem thus reduced is handled in the arrays osource,
409  urn, oweights and osample of size colors2.
410 
411  The sampling proceeds by either of two methods: simulating urn experiment,
412  or conditional method followed by Metropolis-Hastings sampling.
413 
414  Simulating the urn experiment is simply taking one ball at a time, requiring
415  n uniform random variates. The problem is reduced whenever a color has been
416  exhausted.
417 
418  The conditional method divides the colors into groups where the number of
419  balls in each group is determined by sampling from the marginal distribution
420  which is approximated by the univariate Wallenius distribution. Each group
421  is then subdivided by sampling one color at a time until all colors have
422  been sampled.
423 
424  The sample from the conditional method does not have the exact distribution,
425  but it is used as a starting point for the Metropolis-Hastings sampling,
426  which proceeds as follows: colors c1 and c2 are re-sampled using the
427  univariate Wallenius distribution, keeping the samples of all other colors
428  constant. The new sample is accepted or the old sample retained, according
429  to the Metropolis formula which corrects for the slight error introduced
430  by not using the true conditional distribution. c1 and c2 are rotated in
431  an order determined by the variance of each color. This rotation (scan) is
432  repeated nHastings times.
433  */
434 
435  // variables
436  int order1[MAXCOLORS]; // sort order, index into source and destination
437  int order2[MAXCOLORS]; // corresponding index into arrays when equal weights pooled together
438  int order3[MAXCOLORS]; // secondary index for sorting by variance
439  int32_t osource[MAXCOLORS]; // contents of source, sorted by weight with equal weights pooled together
440  int32_t urn[MAXCOLORS]; // balls from osource not taken yet
441  int32_t osample[MAXCOLORS]; // balls sampled
442  double oweights[MAXCOLORS]; // sorted list of weights
443  double wcum[MAXCOLORS]; // list of accumulated probabilities
444  double var[MAXCOLORS]; // sorted list of variance
445  double w = 0.; // weight of balls of one color
446  double w1, w2; // odds within group; mean weight in group
447  double wsum; // total weight of all balls of several or all colors
448  double p; // probability
449  double f0, f1; // multivariate probability function
450  double g0, g1; // conditional probability function
451  double r1, r2; // temporaries in calculation of variance
452  int32_t nn; // number of balls left to sample
453  int32_t m; // number of balls of one color
454  int32_t msum; // total number of balls of several or all colors
455  int32_t N; // total number of balls with nonzero weight
456  int32_t x0, x = 0; // sample of one color
457  int32_t n1, n2, ng; // size of weight group sample or partial sample
458  int32_t m1, m2; // size of weight group
459  int i, j, k; // loop counters
460  int c, c1, c2; // color index
461  int colors2; // reduced number of colors
462  int a, b; // color index delimiting weight group
463  int nHastings; // number of scans in Metropolis-Hastings sampling
464 
465  // check validity of parameters
466  if (n < 0 || colors < 0 || colors > MAXCOLORS) FatalError("Parameter out of range in function MultiWalleniusNCHyp");
467  if (colors == 0) return;
468  if (n == 0) {
469  for (i=0; i<colors; i++) destination[i] = 0; return;
470  }
471 
472  // check validity of array parameters
473  for (i=0, msum=0; i < colors; i++) {
474  m = source[i]; w = weights[i];
475  if (m < 0 || w < 0) FatalError("Parameter negative in function MultiWalleniusNCHyp");
476  if (w) msum += m;
477  }
478  N = msum;
479 
480  // sort colors by weight, heaviest first
481  for (i=0; i < colors; i++) order1[i] = order3[i] = i;
482  for (i=0; i < colors-1; i++) {
483  c = order1[i]; k = i;
484  w = weights[c];
485  if (source[c]==0) w = 0; // zero number treated as zero weight
486  for (j=i+1; j < colors; j++) {
487  c2 = order1[j];
488  if (weights[c2] > w && source[c2]) {
489  w = weights[c2]; k = j;
490  }
491  }
492  order1[i] = order1[k]; order1[k] = c;
493  }
494 
495  // skip any colors with zero weight or zero number.
496  // this solves all problems with zero weights
497  while (colors && (weights[c=order1[colors-1]]==0 || source[c]==0)) {
498  colors--; destination[c] = 0;
499  }
500 
501  // check if there are more than n balls with nonzero weight
502  if (n >= N) {
503  if (n > N) FatalError("Taking more items than there are in function MultiWalleniusNCHyp");
504  for (i = 0; i < colors; i++) {c = order1[i]; destination[c] = source[c];}
505  return;
506  }
507 
508  // copy source and weights into ordered lists
509  // and pool together colors with same weight
510  for (i=0, c2=-1; i < colors; i++) {
511  c = order1[i];
512  if (i==0 || weights[c] != w) {
513  c2++;
514  x = source[c];
515  oweights[c2] = w = weights[c];
516  }
517  else {
518  x += source[c]; // join colors with same weight
519  }
520  urn[c2] = osource[c2] = x;
521  order2[i] = c2;
522  osample[c2] = 0;
523  }
524  colors2 = c2 + 1;
525 
526  // check number of colors left
527  if (colors2 < 3) {
528  // simple cases
529  if (colors2 == 1) osample[0] = n;
530  if (colors2 == 2) {
531  x = WalleniusNCHyp(n, osource[0], N, oweights[0]/oweights[1]);
532  osample[0] = x; osample[1] = n - x;
533  }
534  }
535  else {
536 
537  // more than 2 colors
538  nn = n;
539 
540  // decide which method to use
541  if (nn < 5000 * colors2) {
542 
543  // Simulate urn experiment
544 
545  // Make list of accumulated probabilities of each color
546  for (i=0, wsum=0; i < colors2; i++) {
547  wsum += urn[i] * oweights[i];
548  wcum[i] = wsum;
549  }
550 
551  // take one item nn times
552  j = colors2-1;
553  do {
554  // get random color according to probability distribution wcum
555  p = Random() * wcum[colors2-1];
556  // get color from search in probability distribution wcum
557  for (i=0; i < j; i++) {
558  if (p < wcum[i]) break;
559  }
560 
561  // sample one ball of color i
562  osample[i]++; urn[i]--; nn--;
563 
564  // check if this color has been exhausted
565  if (urn[i] == 0) {
566  if (i != j) {
567  // put exhausted color at the end of lists so that colors2 can be reduced
568  m = osource[i]; osource[i] = osource[j]; osource[j] = m;
569  m = urn[i]; urn[i] = urn[j]; urn[j] = m;
570  m = osample[i]; osample[i] = osample[j]; osample[j] = m;
571  w = oweights[i]; oweights[i] = oweights[j]; oweights[j] = w;
572  // update order2 list (no longer sorted by weight)
573  for (k=0; k<colors; k++) {
574  if (order2[k] == i) order2[k] = j; else
575  if (order2[k] == j) order2[k] = i;
576  }
577  }
578  colors2--; j = colors2-1; // decrement number of colors left in urn
579 
580  if (colors2 == 2 && nn > 50) {
581  // two colors left. use univariate distribution for the rest
582  x = WalleniusNCHyp(nn, urn[0], urn[0]+urn[1], oweights[0]/oweights[1]);
583  osample[0] += x;
584  osample[1] += nn - x;
585  break;}
586 
587  if (colors2 == 1) {
588  // only one color left. The rest is deterministic
589  osample[0] += nn;
590  break;
591  }
592 
593  // make sure wcum is re-calculated from beginning
594  i = 0;
595  }
596 
597  // update list of accumulated probabilities
598  wsum = i > 0 ? wcum[i-1] : 0.;
599  for (k=i; k<colors2; k++) {
600  wsum += urn[k] * oweights[k];
601  wcum[k] = wsum;
602  }
603  }
604  while (nn);
605  }
606 
607  else {
608  // use conditional method to make starting point for
609  // Metropolis-Hastings sampling
610 
611  // divide weights into two groups, heavy and light
612  a = 0; b = colors2-1;
613  w = sqrt(oweights[0] * oweights[colors2-1]);
614  do {
615  c = (a + b) / 2;
616  if (oweights[c] > w) a = c; else b = c;
617  }
618  while (b > a + 1);
619  // heavy group goes from 0 to b-1, light group goes from b to colors2-1
620 
621  // calculate mean weight for heavy color group
622  for (i=0, m1=0, wsum=0; i < b; i++) {
623  m1 += urn[i]; wsum += oweights[i] * urn[i];
624  }
625  w1 = wsum / m1;
626 
627  // calculate mean weight for light color group
628  for (i=b, m2=0, wsum=0; i < colors2; i++) {
629  m2 += urn[i]; wsum += oweights[i] * urn[i];
630  }
631  w2 = wsum / m2;
632 
633  // split partial sample n into heavy (n1) and light (n2)
634  n1 = WalleniusNCHyp(n, m1, m1+m2, w1/w2);
635  n2 = n - n1;
636 
637  // set parameters for first group (heavy)
638  a = 0; ng = n1;
639 
640  // loop twice, for the two groops
641  for (k=0; k < 2; k++) {
642 
643  // split group into single colors by calling univariate distribution b-a-1 times
644  for (i = a; i < b-1; i++) {
645  m = urn[i]; w = oweights[i];
646 
647  // calculate mean weight of remaining colors
648  for (j=i+1, msum=0, wsum=0; j < b; j++) {
649  m1 = urn[j]; w1 = oweights[j];
650  msum += m1; wsum += m1 * w1;
651  }
652 
653  // sample color i in group
654  x = wsum ? WalleniusNCHyp(ng, m, msum + m, w * msum / wsum) : ng;
655 
656  osample[i] = x;
657  ng -= x;
658  }
659 
660  // get the last one in the group
661  osample[i] = ng;
662 
663  // set parameters for second group (light)
664  a = b; b = colors2; ng = n2;
665  }
666 
667  // finished with conditional method.
668  // osample contains starting point for Metropolis-Hastings sampling
669 
670  // make object for calculating probabilities and mean
671  CMultiWalleniusNCHypergeometric wmnc(n, osource, oweights, colors2);
672 
673  wmnc.mean(var); // calculate mean
674  // calculate approximate variance from mean
675  for (i=0; i<colors; i++) {
676  r1 = var[i] * (osource[i]-var[i]);
677  r2 = (n-var[i])*(var[i]+N-n-osource[i]);
678  if (r1 <= 0. || r2 <= 0.) {
679  var[i] = 0.;
680  }
681  else {
682  var[i] = N*r1*r2/((N-1)*(osource[i]*r2+(N-osource[i])*r1));
683  }
684  }
685 
686  // sort again, this time by variance
687  for (i=0; i < colors2-1; i++) {
688  c = order3[i]; k = i;
689  w = var[c];
690  for (j=i+1; j < colors2; j++) {
691  c2 = order3[j];
692  if (var[c2] > w) {
693  w = var[c2]; k = j;
694  }
695  }
696  order3[i] = order3[k]; order3[k] = c;
697  }
698 
699  // number of scans (this value of nHastings has not been fine-tuned)
700  nHastings = 4;
701  if (accuracy < 1E-6) nHastings = 6;
702  if (colors2 > 5) nHastings++;
703 
704  // Metropolis-Hastings sampler
705  f0 = -1.;
706  for (k = 0; k < nHastings; k++) {
707  for (i = 0; i < colors2; i++) {
708  j = i+1;
709  if (j >= colors2) j = 0;
710  c1 = order3[i]; c2 = order3[j];
711  w = oweights[c1] / oweights[c2];
712  n1 = osample[c1] + osample[c2];
713  x0 = osample[c1];
714  x = WalleniusNCHyp(n1, osource[c1], osource[c1]+osource[c2], w);
715  if (x == x0) continue; // accepted
716  if (f0 < 0.) f0 = wmnc.probability(osample);
717  CWalleniusNCHypergeometric nc(n1, osource[c1], osource[c1]+osource[c2], w, accuracy);
718  g0 = nc.probability(x0);
719  g1 = nc.probability(x);
720  osample[c1] = x;
721  osample[c2] = n1 - x;
722  f1 = wmnc.probability(osample);
723  g0 = f1 * g0; g1 = f0 * g1;
724  if (g0 >= g1 || g0 > g1 * Random()) {
725  // new state accepted
726  f0 = -1.;
727  }
728  else {
729  // rejected. restore old sample
730  osample[c1] = x0;
731  osample[c2] = n1 - x0;
732  }
733  }
734  }
735  }
736  }
737 
738  // finished sampling by either method
739  // un-sort sample into destination and untangle re-orderings
740  for (i=0; i < colors; i++) {
741  c1 = order1[i]; c2 = order2[i];
742  if (source[c1] == osource[c2]) {
743  destination[c1] = osample[c2];
744  }
745  else {
746  // split colors with same weight that have been treated as one
747  x = Hypergeometric(osample[c2], source[c1], osource[c2]);
748  destination[c1] = x;
749  osample[c2] -= x;
750  osource[c2] -= source[c1];
751  }
752  }
753 }

789 /******************************************************************************
790 Fisher's noncentral hypergeometric distribution
791 ******************************************************************************/
792 int32_t StochasticLib3::FishersNCHyp (int32_t n, int32_t m, int32_t N, double odds) {
793  /*
794  This function generates a random variate with Fisher's noncentral
795  hypergeometric distribution.
796 
797  This distribution resembles Wallenius noncentral hypergeometric distribution
798  and the two distributions are sometimes confused. A more detailed
799  explanation of this distribution is given below under the multivariate
800  Fisher's noncentral hypergeometric distribution (MultiFishersNCHyp).
801  For further documentation see nchyp.pdf, awailable from www.agner.org/random
802 
803  This function uses inversion by chop-down search from zero when parameters
804  are small, and the ratio-of-uniforms rejection method when the former
805  method would be too slow or would give overflow.
806  */
807  int32_t fak, addd; // used for undoing transformations
808  int32_t x; // result
809 
810  // check if parameters are valid
811  if (n > N || m > N || n < 0 || m < 0 || odds <= 0.) {
812  if (odds == 0.) {
813  if (n > N-m) FatalError("Not enough items with nonzero weight in function FishersNCHyp");
814  return 0;
815  }
816  FatalError("Parameter out of range in function FishersNCHyp");}
817 
818  if (odds == 1.) {
819  // use hypergeometric function if odds == 1
820  return Hypergeometric(n, m, N);
821  }
822 
823  // symmetry transformations
824  fak = 1; addd = 0;
825  if (m > N/2) {
826  // invert m
827  m = N - m;
828  fak = -1; addd = n;
829  }
830 
831  if (n > N/2) {
832  // invert n
833  n = N - n;
834  addd += fak * m; fak = - fak;
835  }
836 
837  if (n > m) {
838  // swap n and m
839  x = n; n = m; m = x;
840  }
841 
842  // cases with only one possible result end here
843  if (n == 0 || odds == 0.) return addd;
844 
845  if (fak == -1) {
846  // reciprocal odds if inverting
847  odds = 1. / odds;
848  }
849 
850  // choose method
851  if (n < 30 && N < 1024 && odds > 1.E-5 && odds < 1.E5) {
852  // use inversion by chop down method
853  x = FishersNCHypInversion (n, m, N, odds);
854  }
855  else {
856  // use ratio-of-uniforms method
857  x = FishersNCHypRatioOfUnifoms (n, m, N, odds);
858  }
859 
860  // undo symmetry transformations
861  return x * fak + addd;
862 }

864 /***********************************************************************
865 Subfunctions used by FishersNCHyp
866 ***********************************************************************/
867 
868 int32_t StochasticLib3::FishersNCHypInversion (int32_t n, int32_t m, int32_t N, double odds) {
869  /*
870  Subfunction for FishersNCHyp distribution.
871  Implements Fisher's noncentral hypergeometric distribution by inversion
872  method, using chop-down search starting at zero.
873 
874  Valid only for 0 <= n <= m <= N/2.
875  Without overflow check the parameters must be limited to n < 30, N < 1024,
876  and 1.E-5 < odds < 1.E5. This limitation is acceptable because this method
877  is slow for higher n.
878 
879  The execution time of this function grows with n.
880 
881  See the file nchyp.pdf for theoretical explanation.
882  */
883  int32_t x; // x value
884  int32_t L; // derived parameter
885  double f; // scaled function value
886  double sum; // scaled sum of function values
887  double a1, a2, b1, b2, f1, f2; // factors in recursive calculation
888  double u; // uniform random variate
889 
890  L = N - m - n;
891 
892  if (n != fnc_n_last || m != fnc_m_last || N != fnc_N_last || odds != fnc_o_last) {
893  // parameters have changed. set-up
894  fnc_n_last = n; fnc_m_last = m; fnc_N_last = N; fnc_o_last = odds;
895 
896  // f(0) is set to an arbitrary value because it cancels out.
897  // A low value is chosen to avoid overflow.
898  fnc_f0 = 1.E-100;
899 
900  // calculate summation of e(x), using the formula:
901  // f(x) = f(x-1) * (m-x+1)*(n-x+1)*odds / (x*(L+x))
902  // All divisions are avoided by scaling the parameters
903  sum = f = fnc_f0; fnc_scale = 1.;
904  a1 = m; a2 = n; b1 = 1; b2 = L + 1;
905  for (x = 1; x <= n; x++) {
906  f1 = a1 * a2 * odds;
907  f2 = b1 * b2;
908  a1--; a2--; b1++; b2++;
909  f *= f1;
910  sum *= f2;
911  fnc_scale *= f2;
912  sum += f;
913  // overflow check. not needed if parameters are limited:
914  // if (sum > 1E100) {sum *= 1E-100; f *= 1E-100; fnc_scale *= 1E-100;}
915  }
916  fnc_f0 *= fnc_scale;
917  fnc_scale = sum;
918  // now f(0) = fnc_f0 / fnc_scale.
919  // We are still avoiding all divisions by saving the scale factor
920  }
921 
922  // uniform random
923  u = Random() * fnc_scale;
924 
925  // recursive calculation:
926  // f(x) = f(x-1) * (m-x+1)*(n-x+1)*odds / (x*(L+x))
927  f = fnc_f0; x = 0; a1 = m; a2 = n; b1 = 0; b2 = L;
928  do {
929  u -= f;
930  if (u <= 0) break;
931  x++; b1++; b2++;
932  f *= a1 * a2 * odds;
933  u *= b1 * b2;
934  // overflow check. not needed if parameters are limited:
935  // if (u > 1.E100) {u *= 1E-100; f *= 1E-100;}
936  a1--; a2--;
937  }
938  while (x < n);
939  return x;
940 }

942 int32_t StochasticLib3::FishersNCHypRatioOfUnifoms (int32_t n, int32_t m, int32_t N, double odds) {
943  /*
944  Subfunction for FishersNCHyp distribution.
945  Valid for 0 <= n <= m <= N/2, odds != 1
946 
947  Fisher's noncentral hypergeometric distribution by ratio-of-uniforms
948  rejection method.
949 
950  The execution time of this function is almost independent of the parameters.
951  */
952  int32_t L; // N-m-n
953  int32_t mode; // mode
954  double mean; // mean
955  double variance; // variance
956  double x; // real sample
957  int32_t k; // integer sample
958  double u; // uniform random
959  double lf; // ln(f(x))
960  double AA, BB, g1, g2; // temporary
961 
962  L = N - m - n;
963 
964  if (n != fnc_n_last || m != fnc_m_last || N != fnc_N_last || odds != fnc_o_last) {
965  // parameters have changed. set-up
966  fnc_n_last = n; fnc_m_last = m; fnc_N_last = N; fnc_o_last = odds;
967 
968  // find approximate mean
969  AA = (m+n)*odds+L; BB = sqrt(AA*AA - 4*odds*(odds-1)*m*n);
970  mean = (AA-BB)/(2*(odds-1));
971 
972  // find approximate variance
973  AA = mean * (m-mean); BB = (n-mean)*(mean+L);
974  variance = N*AA*BB/((N-1)*(m*BB+(n+L)*AA));
975 
976  // compute log(odds)
977  fnc_logb = log(odds);
978 
979  // find center and width of hat function
980  fnc_a = mean + 0.5;
981  fnc_h = 1.028 + 1.717*sqrt(variance+0.5) + 0.032*fabs(fnc_logb);
982 
983  // find safety bound
984  fnc_bound = (int32_t)(mean + 4.0 * fnc_h);
985  if (fnc_bound > n) fnc_bound = n;
986 
987  // find mode
988  mode = (int32_t)(mean);
989  g1 =(double)(m-mode)*(n-mode)*odds;
990  g2 =(double)(mode+1)*(L+mode+1);
991  if (g1 > g2 && mode < n) mode++;
992 
993  // value at mode to scale with:
994  fnc_lfm = mode * fnc_logb - fc_lnpk(mode, L, m, n);
995  }
996 
997  while(1) {
998  u = Random();
999  if (u == 0) continue; // avoid divide by 0
1000  x = fnc_a + fnc_h * (Random()-0.5)/u;
1001  if (x < 0. || x > 2E9) continue; // reject, avoid overflow
1002  k = (int32_t)(x); // truncate
1003  if (k > fnc_bound) continue; // reject if outside safety bound
1004  lf = k*fnc_logb - fc_lnpk(k,L,m,n) - fnc_lfm;// compute function value
1005  if (u * (4.0 - u) - 3.0 <= lf) break; // lower squeeze accept
1006  if (u * (u-lf) > 1.0) continue; // upper squeeze reject
1007  if (2.0 * log(u) <= lf) break; // final acceptance
1008  }
1009  return k;
1010 }

1013 /***********************************************************************
1014 Multivariate Fisher's noncentral hypergeometric distribution
1015 ***********************************************************************/
1016 void StochasticLib3::MultiFishersNCHyp (int32_t * destination,
1017 int32_t * source, double * weights, int32_t n, int colors) {
1018  /*
1019  This function generates a vector of random variates with the
1020  multivariate Fisher's noncentral hypergeometric distribution.
1021 
1022  This distribution is defined as the conditional distribution of 'colors'
1023  independent binomial variates
1024  x[i] = binomial(source[i], p[i])
1025  on the condition that the sum of all x[i] is n.
1026  p[i] = r * weights[i] / (1 + r * weights[i]),
1027  r is an arbitrary scale factor.
1028 
1029  Parameters:
1030  destination: An output array to receive the number of balls of each
1031  color. Must have space for at least 'colors' elements.
1032  source: An input array containing the number of balls of each
1033  color in the urn. Must have 'colors' elements.
1034  All elements must be non-negative.
1035  weights: The odds of each color. Must have 'colors' elements.
1036  All elements must be non-negative.
1037  n: The number of balls drawn from the urn.
1038  Can't exceed the total number of balls with nonzero weight
1039  in the urn.
1040  colors: The number of possible colors.
1041 
1042  Method: The conditional method is used for generating a sample with the
1043  approximate distribution. This sample is used as a starting point for
1044  a Gibbs sampler. The accuracy depends on the number of scans with the
1045  Gibbs sampler.
1046 
1047  The function will reduce the number of colors, if possible, by eliminating
1048  colors with zero weight or zero number and pooling together colors with the
1049  same weight. A symmetry transformation is used if more than half the balls
1050  are taken. The problem thus reduced is handled in the arrays osource,
1051  oweights and osample of dimension colors2.
1052  */
1053  int order1[MAXCOLORS]; // sort order, index into source and destination
1054  int order2[MAXCOLORS]; // corresponding index into osource when equal weights pooled together
1055  int order3[MAXCOLORS]; // secondary index for sorting by variance
1056  int32_t osource[MAXCOLORS]; // contents of source, sorted by weight with equal weights pooled together
1057  int32_t osample[MAXCOLORS]; // balls sampled, sorted by weight
1058  double oweights[MAXCOLORS]; // sorted list of weights
1059  double var[MAXCOLORS]; // sorted list of variance
1060  int32_t x = 0; // univariate sample
1061  int32_t m; // number of items of one color
1062  int32_t m1, m2; // number of items in each weight group
1063  int32_t msum; // total number of items of several or all colors
1064  int32_t n0; // remaining balls to sample
1065  int32_t n1, n2; // sample size for each weight group
1066  double w = 0.; // weight or variance of items of one color
1067  double w1, w2; // mean weight of each weight group
1068  double wsum; // total weight of all items of several or all colors
1069  double odds; // weight ratio
1070  int i, j, k; // loop counters
1071  int a, b; // limits for weight group
1072  int c, c1, c2; // color index
1073  int colors2; // reduced number of colors, number of entries in osource
1074  int ngibbs; // number of scans in Gibbs sampler
1075  int invert = 0; // 1 if symmetry transformation used
1076 
1077  // check validity of parameters
1078  if (n < 0 || colors < 0 || colors > MAXCOLORS) FatalError("Parameter out of range in function MultiFishersNCHyp");
1079  if (colors == 0) return;
1080  if (n == 0) {
1081  for (i=0; i<colors; i++) destination[i] = 0; return;
1082  }
1083 
1084  // check validity of array parameters
1085  for (i=0, msum=0; i < colors; i++) {
1086  m = source[i]; w = weights[i];
1087  if (m < 0 || w < 0) FatalError("Parameter negative in function MultiFishersNCHyp");
1088  if (w) msum += m;
1089  }
1090 
1091  // sort by weight, heaviest first
1092  for (i=0; i < colors; i++) order1[i] = order3[i] = i;
1093  for (i=0; i < colors-1; i++) {
1094  c = order1[i]; k = i;
1095  w = weights[c]; if (source[c]==0) w = 0;
1096  for (j=i+1; j < colors; j++) {
1097  c2 = order1[j];
1098  if (weights[c2] > w && source[c2]) {
1099  w = weights[c2]; k = j;
1100  }
1101  }
1102  order1[i] = order1[k]; order1[k] = c;
1103  }
1104 
1105  // Skip any items with zero weight
1106  // this solves all problems with zero weights
1107  while (colors && (weights[c=order1[colors-1]]==0 || source[c]==0)) {
1108  colors--; destination[c] = 0;
1109  }
1110 
1111  // check if we are taking all, or too many, balls
1112  if (n >= msum) {
1113  if (n > msum) FatalError("Taking more items than there are in function MultiFishersNCHyp");
1114  for (i = 0; i < colors; i++) {c = order1[i]; destination[c] = source[c];}
1115  return;
1116  }
1117 
1118  if (n > msum / 2) {
1119  // improve accuracy by symmetry transformation
1120  for (i=0, j=colors-1; i < j; i++, j--) { // reverse order list
1121  c = order1[i]; order1[i] = order1[j]; order1[j] = c;}
1122  n = msum - n; invert = 1;
1123  }
1124 
1125  // copy source and weights into ordered lists and pool together colors with same weight
1126  for (i=0, c2=-1; i < colors; i++) {
1127  c = order1[i];
1128  if (i==0 || weights[c] != w) {
1129  c2++;
1130  x = source[c];
1131  oweights[c2] = w = invert ? 1./weights[c] : weights[c];
1132  }
1133  else {
1134  x += source[c];
1135  }
1136  osource[c2] = x;
1137  order2[i] = c2;
1138  osample[c2] = 0;
1139  }
1140  colors2 = c2 + 1;
1141 
1142  // simple cases
1143  if (colors2 == 1) osample[0] = n;
1144  if (colors2 == 2) {
1145  x = FishersNCHyp(n, osource[0], msum, oweights[0]/oweights[1]);
1146  osample[0] = x; osample[1] = n - x;
1147  }
1148 
1149  if (colors2 > 2) {
1150  // divide weights into two groups, heavy and light
1151  a = 0; b = colors2-1;
1152  w = sqrt(oweights[0] * oweights[colors2-1]);
1153  do {
1154  c = (a + b) / 2;
1155  if (oweights[c] > w) a = c; else b = c;
1156  }
1157  while (b > a + 1);
1158  a = 0; // heavy group goes from a to b-1, light group goes from b to colors2-1
1159 
1160  // calculate mean weight for heavy group
1161  for (i=a, m1=0, wsum=0; i < b; i++) {
1162  m1 += osource[i]; wsum += oweights[i] * osource[i];
1163  }
1164  w1 = wsum / m1;
1165 
1166  // calculate mean weight for light group
1167  for (i=b, m2=0, wsum=0; i < colors2; i++) {
1168  m2 += osource[i]; wsum += oweights[i] * osource[i];
1169  }
1170  w2 = wsum / m2;
