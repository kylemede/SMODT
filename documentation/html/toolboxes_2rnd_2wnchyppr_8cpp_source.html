342 double CWalleniusNCHypergeometric::mean(void) {
343  // find approximate mean
344  int iter; // number of iterations
345  double a, b; // temporaries in calculation of first guess
346  double mean, mean1; // iteration value of mean
347  double m1r, m2r; // 1/m, 1/m2
348  double e1, e2; // temporaries
349  double g; // function to find root of
350  double gd; // derivative of g
351  double omegar; // 1/omega
352 
353  if (omega == 1.) { // simple hypergeometric
354  return (double)(m)*n/N;
355  }
356 
357  // calculate Cornfield mean of Fisher noncentral hypergeometric distribution as first guess
358  a = (m+n)*omega + (N-m-n);
359  b = a*a - 4.*omega*(omega-1.)*m*n;
360  b = b > 0. ? sqrt(b) : 0.;
361  mean = (a-b)/(2.*(omega-1.));
362  if (mean < xmin) mean = xmin;
363  if (mean > xmax) mean = xmax;
364 
365  m1r = 1./m; m2r = 1./(N-m);
366  iter = 0;
367 
368  if (omega > 1.) {
369  do { // Newton Raphson iteration
370  mean1 = mean;
371  e1 = 1.-(n-mean)*m2r;
372  if (e1 < 1E-14) {
373  e2 = 0.; // avoid underflow
374  }
375  else {
376  e2 = pow(e1,omega-1.);
377  }
378  g = e2*e1 + (mean-m)*m1r;
379  gd = e2*omega*m2r + m1r;
380  mean -= g / gd;
381  if (mean < xmin) mean = xmin;
382  if (mean > xmax) mean = xmax;
383  if (++iter > 40) {
384  FatalError("Search for mean failed in function CWalleniusNCHypergeometric::mean");
385  }
386  }
387  while (fabs(mean1 - mean) > 2E-6);
388  }
389  else { // omega < 1
390  omegar = 1./omega;
391  do { // Newton Raphson iteration
392  mean1 = mean;
393  e1 = 1.-mean*m1r;
394  if (e1 < 1E-14) {
395  e2 = 0.; // avoid underflow
396  }
397  else {
398  e2 = pow(e1,omegar-1.);
399  }
400  g = 1.-(n-mean)*m2r-e2*e1;
401  gd = e2*omegar*m1r + m2r;
402  mean -= g / gd;
403  if (mean < xmin) mean = xmin;
404  if (mean > xmax) mean = xmax;
405  if (++iter > 40) {
406  FatalError("Search for mean failed in function CWalleniusNCHypergeometric::mean");
407  }
408  }
409  while (fabs(mean1 - mean) > 2E-6);
410  }
411  return mean;
412 }

604  // The return value will be 1 with probability f(x_)/h and 0 with probability
605  // 1-f(x_)/h.
606  // This is equivalent to calling sto->Bernouilli(probability(x_)/h),
607  // but this method is faster. The method used here avoids calculating the
608  // Wallenius probability by sampling in the t-domain.
609  // rh is a uniform random number in the interval 0 <= rh < h. The function
610  // uses additional random numbers generated from sto.
611  // This function is intended for use in rejection methods for sampling from
612  // the Wallenius distribution. It is called from
613  // StochasticLib3::WalleniusNCHypRatioOfUnifoms in the file stoc3.cpp
614  double f0; // Lambda*Phi()
615  double phideri0; // phi()/rd
616  double qi; // 2^(-r*omega[i])
617  double qi1; // 1-qi
618  double omegai[2] = {omega,1.}; // weights for each color
619  double romegi; // r*omega[i]
620  double xi[2] = {x_, n-x_}; // number of each color sampled
621  double k; // adjusted width for majorizing function Ypsilon(t)
622  double erfk; // erf correction
623  double rdm1; // rd - 1
624  double G_integral; // integral of majorizing function Ypsilon(t)
625  double ts; // t sampled from Ypsilon(t) distribution
626  double logts; // log(ts)
627  double rlogts; // r*log(ts)
628  double fts; // Phi(ts)/rd
629  double rgts; // 1/(Ypsilon(ts)/rd)
630  double t2; // temporary in calculation of Ypsilon(ts)
631  int i, j; // loop counters
632  static const double rsqrt8 = 0.3535533905932737622; // 1/sqrt(8)
633  static const double sqrt2pi = 2.506628274631000454; // sqrt(2*pi)
634 
635  x = x_; // save x in class object
636  lnbico(); // calculate bico = log(Lambda)
637  findpars(); // calculate r, d, rd, w, E
638  if (E > 0.) {
639  k = log(E); // correction for majorizing function
640  k = 1. + 0.0271 * (k * sqrt(k));
641  }
642  else k = 1.;
643  k *= w; // w * k
644  rdm1 = rd - 1.;
645 
646  // calculate phi()/rd
647  phideri0 = -LN2 * rdm1;
648  for (i=0; i<2; i++) {
649  romegi = r * omegai[i];
650  if (romegi > 40.) {
651  qi=0.; qi1 = 1.; // avoid underflow
652  }
653  else {
654  qi1 = pow2_1(-romegi, &qi);
655  }
656  phideri0 += xi[i] * log1mx(qi, qi1);
657  }
658 
659  erfk = Erf(rsqrt8 / k);
660  f0 = rd * exp(phideri0 + bico);
661  G_integral = f0 * sqrt2pi * k * erfk;
662 
663  if (G_integral <= h) { // G fits under h-hat
664  do {
665  ts = sto->Normal(0,k); // sample ts from normal distribution
666  }
667  while (fabs(ts) >= 0.5); // reject values outside interval, and avoid ts = 0
668  ts += 0.5; // ts = normal distributed in interval (0,1)
669 
670  for (fts=0., j=0; j<2; j++) { // calculate (Phi(ts)+Phi(1-ts))/2
671  logts = log(ts); rlogts = r * logts; // (ts = 0 avoided above)
672  fts += exp(log1pow(rlogts*omega,xi[0]) + log1pow(rlogts,xi[1]) + rdm1*logts + bico);
673  ts = 1. - ts;
674  }
675  fts *= 0.5;
676 
677  t2 = (ts-0.5) / k; // calculate 1/Ypsilon(ts)
678  rgts = exp(-(phideri0 + bico - 0.5 * t2*t2));
679  return rh < G_integral * fts * rgts; // Bernouilli variate
680  }
681 
682  else { // G > h: can't use sampling in t-domain
683  return rh < probability(x);
684  }

692 double CWalleniusNCHypergeometric::recursive() {
693  // recursive calculation
694  // Wallenius noncentral hypergeometric distribution by recursion formula
695  // Approximate by ignoring probabilities < accuracy and minimize storage requirement
696  const int BUFSIZE = 512; // buffer size
697  double p[BUFSIZE+2]; // probabilities
698  double * p1, * p2; // offset into p
699  double mxo; // (m-x)*omega
700  double Nmnx; // N-m-nu+x
701  double y, y1; // save old p[x] before it is overwritten
702  double d1, d2, dcom; // divisors in probability formula
703  double accuracya; // absolute accuracy
704  int32_t xi, nu; // xi, nu = recursion values of x, n
705  int32_t x1, x2; // xi_min, xi_max
706 
707  accuracya = 0.005f * accuracy; // absolute accuracy
708  p1 = p2 = p + 1; // make space for p1[-1]
709  p1[-1] = 0.; p1[0] = 1.; // initialize for recursion
710  x1 = x2 = 0;
711  for (nu=1; nu<=n; nu++) {
712  if (n - nu < x - x1 || p1[x1] < accuracya) {
713  x1++; // increase lower limit when breakpoint passed or probability negligible
714  p2--; // compensate buffer offset in order to reduce storage space
715  }
716  if (x2 < x && p1[x2] >= accuracya) {
717  x2++; y1 = 0.; // increase upper limit until x has been reached
718  }
719  else {
720  y1 = p1[x2];
721  }
722  if (x1 > x2) return 0.;
723  if (p2+x2-p > BUFSIZE) FatalError("buffer overrun in function CWalleniusNCHypergeometric::recursive");
724 
725  mxo = (m-x2)*omega;
726  Nmnx = N-m-nu+x2+1;
727  for (xi = x2; xi >= x1; xi--) { // backwards loop
728  d2 = mxo + Nmnx;
729  mxo += omega; Nmnx--;
730  d1 = mxo + Nmnx;
731  dcom = 1. / (d1 * d2); // save a division by making common divisor
732  y = p1[xi-1]*mxo*d2*dcom + y1*(Nmnx+1)*d1*dcom;
733  y1 = p1[xi-1]; // (warning: pointer alias, can't swap instruction order)
734  p2[xi] = y;
735  }
736  p1 = p2;
737  }
738 
739  if (x < x1 || x > x2) return 0.;
740  return p1[x];

774 double CWalleniusNCHypergeometric::laplace() {
775  // Laplace's method with narrow integration interval,
776  // using error function residues table, defined in erfres.cpp
777  // Note that this function can only be used when the integrand peak is narrow.
778  // findpars() must be called before this function.
779 
780  const int COLORS = 2; // number of colors
781  const int MAXDEG = 40; // arraysize, maximum expansion degree
782  int degree; // max expansion degree
783  double accur; // stop expansion when terms below this threshold
784  double omegai[COLORS] = {omega, 1.}; // weights for each color
785  double xi[COLORS] = {x, n-x}; // number of each color sampled
786  double f0; // factor outside integral
787  double rho[COLORS]; // r*omegai
788  double qi; // 2^(-rho)
789  double qi1; // 1-qi
790  double qq[COLORS]; // qi / qi1
791  double eta[COLORS+1][MAXDEG+1]; // eta coefficients
792  double phideri[MAXDEG+1]; // derivatives of phi
793  double PSIderi[MAXDEG+1]; // derivatives of PSI
794  double * erfresp; // pointer to table of error function residues
795 
796  // variables in asymptotic summation
797  static const double sqrt8 = 2.828427124746190098; // sqrt(8)
798  double qqpow; // qq^j
799  double pow2k; // 2^k
800  double bino; // binomial coefficient
801  double vr; // 1/v, v = integration interval
802  double v2m2; // (2*v)^(-2)
803  double v2mk1; // (2*v)^(-k-1)
804  double s; // summation term
805  double sum; // Taylor sum
806 
807  int i; // loop counter for color
808  int j; // loop counter for derivative
809  int k; // loop counter for expansion degree
810  int ll; // k/2
811  int converg = 0; // number of consequtive terms below accuracy
812  int PrecisionIndex; // index into ErfRes table according to desired precision
813 
814  // initialize
815  for (k = 0; k <= 2; k++) phideri[k] = PSIderi[k] = 0;
816 
817  // find rho[i], qq[i], first eta coefficients, and zero'th derivative of phi
818  for (i = 0; i < COLORS; i++) {
819  rho[i] = r * omegai[i];
820  if (rho[i] > 40.) {
821  qi=0.; qi1 = 1.;} // avoid underflow
822  else {
823  qi1 = pow2_1(-rho[i], &qi);} // qi=2^(-rho), qi1=1.-2^(-rho)
824  qq[i] = qi / qi1; // 2^(-r*omegai)/(1.-2^(-r*omegai))
825  // peak = zero'th derivative
826  phideri[0] += xi[i] * log1mx(qi, qi1);
827  // eta coefficients
828  eta[i][0] = 0.;
829  eta[i][1] = eta[i][2] = rho[i]*rho[i];
830  }
831 
832  // r, rd, and w must be calculated by findpars()
833  // zero'th derivative
834  phideri[0] -= (rd - 1.) * LN2;
835  // scaled factor outside integral
836  f0 = rd * exp(phideri[0] + lnbico());
837 
838  vr = sqrt8 * w;
839  phideri[2] = phi2d;
840 
841  // get table according to desired precision
842  PrecisionIndex = (-FloorLog2((float)accuracy) - ERFRES_B + ERFRES_S - 1) / ERFRES_S;
843  if (PrecisionIndex < 0) PrecisionIndex = 0;
844  if (PrecisionIndex > ERFRES_N-1) PrecisionIndex = ERFRES_N-1;
845  while (w * NumSDev[PrecisionIndex] > 0.3) {
846  // check if integration interval is too wide
847  if (PrecisionIndex == 0) {
848  FatalError("Laplace method failed. Peak width too high in function CWalleniusNCHypergeometric::laplace");
849  break;}
850  PrecisionIndex--; // reduce precision to keep integration interval narrow
851  }
852  erfresp = ErfRes[PrecisionIndex]; // choose desired table
853 
854  degree = MAXDEG; // max expansion degree
855  if (degree >= ERFRES_L*2) degree = ERFRES_L*2-2;
856 
857  // set up for starting loop at k=3
858  v2m2 = 0.25 * vr * vr; // (2*v)^(-2)
859  PSIderi[0] = 1.;
860  pow2k = 8.;
861  sum = 0.5 * vr * erfresp[0];
862  v2mk1 = 0.5 * vr * v2m2 * v2m2;
863  accur = accuracy * sum;
864 
865  // summation loop
866  for (k = 3; k <= degree; k++) {
867  phideri[k] = 0.;
868 
869  // loop for all (2) colors
870  for (i = 0; i < COLORS; i++) {
871  eta[i][k] = 0.;
872  // backward loop for all powers
873  for (j = k; j > 0; j--) {
874  // find coefficients recursively from previous coefficients
875  eta[i][j] = eta[i][j]*(j*rho[i]-(k-2)) + eta[i][j-1]*rho[i]*(j-1);
876  }
877  qqpow = 1.;
878  // forward loop for all powers
879  for (j=1; j<=k; j++) {
880  qqpow *= qq[i]; // qq^j
881  // contribution to derivative
882  phideri[k] += xi[i] * eta[i][j] * qqpow;
883  }
884  }
885 
886  // finish calculation of derivatives
887  phideri[k] = -pow2k*phideri[k] + 2*(1-k)*phideri[k-1];
888 
889  pow2k *= 2.; // 2^k
890 
891  // loop to calculate derivatives of PSI from derivatives of psi.
892  // terms # 0, 1, 2, k-2, and k-1 are zero and not included in loop.
893  // The j'th derivatives of psi are identical to the derivatives of phi for j>2, and
894  // zero for j=1,2. Hence we are using phideri[j] for j>2 here.
895  PSIderi[k] = phideri[k]; // this is term # k
896  bino = 0.5 * (k-1) * (k-2); // binomial coefficient for term # 3
897  for (j = 3; j < k-2; j++) { // loop for remaining nonzero terms (if k>5)
898  PSIderi[k] += PSIderi[k-j] * phideri[j] * bino;
899  bino *= double(k-j)/double(j);
900  }
901  if ((k & 1) == 0) { // only for even k
902  ll = k/2;
903  s = PSIderi[k] * v2mk1 * erfresp[ll];
904  sum += s;
905 
906  // check for convergence of Taylor expansion
907  if (fabs(s) < accur) converg++; else converg = 0;
908  if (converg > 1) break;
909 
910  // update recursive expressions
911  v2mk1 *= v2m2;
912  }
913  }
914  // multiply by terms outside integral
915  return f0 * sum;
916 }

919 double CWalleniusNCHypergeometric::integrate() {
920  // Wallenius non-central hypergeometric distribution function
921  // calculation by numerical integration with variable-length steps
922  // NOTE: findpars() must be called before this function.
923  double s; // result of integration step
924  double sum; // integral
925  double ta, tb; // subinterval for integration step
926 
927  lnbico(); // compute log of binomial coefficients
928 
929  // choose method:
930  if (w < 0.02 || w < 0.1 && (x==m || n-x==N-m) && accuracy > 1E-6) {
931  // normal method. Step length determined by peak width w
932  double delta, s1;
933  s1 = accuracy < 1E-9 ? 0.5 : 1.;
934  delta = s1 * w; // integration steplength
935  ta = 0.5 + 0.5 * delta;
936  sum = integrate_step(1.-ta, ta); // first integration step around center peak
937  do {
938  tb = ta + delta;
939  if (tb > 1.) tb = 1.;
940  s = integrate_step(ta, tb); // integration step to the right of peak
941  s += integrate_step(1.-tb,1.-ta); // integration step to the left of peak
942  sum += s;
943  if (s < accuracy * sum) break; // stop before interval finished if accuracy reached
944  ta = tb;
945  if (tb > 0.5 + w) delta *= 2.; // increase step length far from peak
946  }
947  while (tb < 1.);
948  }
949  else {
950  // difficult situation. Step length determined by inflection points
951  double t1, t2, tinf, delta, delta1;
952  sum = 0.;
953  // do left and right half of integration interval separately:
954  for (t1=0., t2=0.5; t1 < 1.; t1+=0.5, t2+=0.5) {
955  // integrate from 0 to 0.5 or from 0.5 to 1
956  tinf = search_inflect(t1, t2); // find inflection point
957  delta = tinf - t1; if (delta > t2 - tinf) delta = t2 - tinf; // distance to nearest endpoint
958  delta *= 1./7.; // 1/7 will give 3 steps to nearest endpoint
959  if (delta < 1E-4) delta = 1E-4;
960  delta1 = delta;
961  // integrate from tinf forwards to t2
962  ta = tinf;
963  do {
964  tb = ta + delta1;
965  if (tb > t2 - 0.25*delta1) tb = t2; // last step of this subinterval
966  s = integrate_step(ta, tb); // integration step
967  sum += s;
968  delta1 *= 2; // double steplength
969  if (s < sum * 1E-4) delta1 *= 8.; // large step when s small
970  ta = tb;
971  }
972  while (tb < t2);
973  if (tinf) {
974  // integrate from tinf backwards to t1
975  tb = tinf;
976  do {
977  ta = tb - delta;
978  if (ta < t1 + 0.25*delta) ta = t1; // last step of this subinterval
979  s = integrate_step(ta, tb); // integration step
980  sum += s;
981  delta *= 2; // double steplength
982  if (s < sum * 1E-4) delta *= 8.; // large step when s small
983  tb = ta;}
984  while (ta > t1);
985  }
986  }
987  }
988  return sum * rd;
989 }

1044 double CWalleniusNCHypergeometric::search_inflect(double t_from, double t_to) {

1046  // t_from < t < t_to

1642  // calculation by numerical integration with variable-length steps

1662  s += integrate_step(1.-tb,1.-ta); // integration step to the left of peak
1663  sum += s;

1665  ta = tb;
1666  if (tb > 0.5 + w) delta *= 2.; // increase step length far from peak
1667  }
1668  while (tb < 1.);

1670 
1671  else {
1672  // difficult situation. Step length determined by inflection points
1673  double t1, t2, tinf, delta, delta1;
1674  sum = 0.;

1676  for (t1=0., t2=0.5; t1 < 1.; t1+=0.5, t2+=0.5) {

1678  tinf = search_inflect(t1, t2); // find inflection point
1679  delta = tinf - t1; if (delta > t2 - tinf) delta = t2 - tinf; // distance to nearest endpoint

1681  if (delta < 1E-4) delta = 1E-4;
1682  delta1 = delta;
1683  // integrate from tinf forwards to t2
1684  ta = tinf;

1686  tb = ta + delta1;

1688  s = integrate_step(ta, tb); // integration step
1689  sum += s;

1692  ta = tb;
1693  }

1705  tb = ta;
1706  }

1708  }

1773 double CMultiWalleniusNCHypergeometric::search_inflect(double t_from, double t_to) {
1774  // search for an inflection point of the integrand PHI(t) in the interval
1775  // t_from < t < t_to
1776  double t, t1; // independent variable
1777  double rho[MAXCOLORS]; // r*omega[i]
1778  double q; // t^rho[i] / (1-t^rho[i])
1779  double q1; // 1-t^rho[i]
1780  double zeta[MAXCOLORS][4][4]; // zeta[i,j,k] coefficients
1781  double phi[4]; // derivatives of phi(t) = log PHI(t)
1782  double Z2; // PHI''(t)/PHI(t)
1783  double Zd; // derivative in Newton Raphson iteration
1784  double rdm1; // r * d - 1
1785  double tr; // 1/t
1786  double log2t; // log2(t)
1787  double method; // 0 for z2'(t) method, 1 for z3(t) method
1788  int i; // color
1789  int iter; // count iterations
1790 
1791  rdm1 = rd - 1.;
1792  if (t_from == 0 && rdm1 <= 1.) return 0.; //no inflection point
1793  t = 0.5 * (t_from + t_to);
1794  for (i = 0; i < colors; i++) { // calculate zeta coefficients
1795  rho[i] = r * omega[i];
1796  zeta[i][1][1] = rho[i];
1797  zeta[i][1][2] = rho[i] * (rho[i] - 1.);
1798  zeta[i][2][2] = rho[i] * rho[i];
1799  zeta[i][1][3] = zeta[i][1][2] * (rho[i] - 2.);
1800  zeta[i][2][3] = zeta[i][1][2] * rho[i] * 3.;
1801  zeta[i][3][3] = zeta[i][2][2] * rho[i] * 2.;
1802  }
1803  iter = 0;
1804 
1805  do {
1806  t1 = t;
1807  tr = 1. / t;
1808  log2t = log(t)*(1./LN2);
1809  phi[1] = phi[2] = phi[3] = 0.;
1810  for (i=0; i<colors; i++) { // calculate first 3 derivatives of phi(t)
1811  if (rho[i] == 0.) continue;
1812  q1 = pow2_1(rho[i]*log2t,&q);
1813  q /= q1;
1814  phi[1] -= x[i] * zeta[i][1][1] * q;
1815  phi[2] -= x[i] * q * (zeta[i][1][2] + q * zeta[i][2][2]);
1816  phi[3] -= x[i] * q * (zeta[i][1][3] + q * (zeta[i][2][3] + q * zeta[i][3][3]));
1817  }
1818  phi[1] += rdm1;
1819  phi[2] -= rdm1;
1820  phi[3] += 2. * rdm1;
1821  phi[1] *= tr;
1822  phi[2] *= tr * tr;
1823  phi[3] *= tr * tr * tr;
1824  method = (iter & 2) >> 1; // alternate between the two methods
1825  Z2 = phi[1]*phi[1] + phi[2];
1826  Zd = method*phi[1]*phi[1]*phi[1] + (2.+method)*phi[1]*phi[2] + phi[3];
1827 
1828  if (t < 0.5) {
1829  if (Z2 > 0) {
1830  t_from = t;
1831  }
1832  else {
1833  t_to = t;
1834  }
1835  if (Zd >= 0) {
1836  // use binary search if Newton-Raphson iteration makes problems
1837  t = (t_from ? 0.5 : 0.2) * (t_from + t_to);
1838  }
1839  else {
1840  // Newton-Raphson iteration
1841  t -= Z2 / Zd;
1842  }
1843  }
1844  else {
1845  if (Z2 < 0) {
1846  t_from = t;
1847  }
1848  else {
1849  t_to = t;
1850  }
1851  if (Zd <= 0) {
1852  // use binary search if Newton-Raphson iteration makes problems
1853  t = 0.5 * (t_from + t_to);
1854  }
1855  else {
1856  // Newton-Raphson iteration
1857  t -= Z2 / Zd;
1858  }
1859  }
1860  if (t >= t_to) t = (t1 + t_to) * 0.5;
1861  if (t <= t_from) t = (t1 + t_from) * 0.5;
1862  if (++iter > 20) FatalError("Search for inflection point failed in function CMultiWalleniusNCHypergeometric::search_inflect");
1863  }
