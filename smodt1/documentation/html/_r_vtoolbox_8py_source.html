161 def subtractPlanetRV(rvDataFilename,e_p,p_p,K_p,argPeri_p,T_p,Tc_p):
162  """
163  This function will subtract the RV due to a companion planet leaving only the residual
164  that could be caused by a secondary star or another planet. If the system you are
165  investigating has a long period companion star, then this is useful to subtract
166  the RV due to the circum-primary star with known orbital elements and focus the simulation
167  on solving only for the companion star. Calculating both for each proposed set of
168  elements of the companion star during the simulation is un-necesarily CPU taxing.
169  """

173  RVs = rvDataDict['RVs']
174 
175  rvWithoutPlanetResiduals = []
176  for RVdataSet in range(0,len(RVs)):
177  print '\nworking on RVdataSet ',RVdataSet
178  for epoch in range(0,len(RV_epochs[RVdataSet])):
179  v_r_p = vrCalculatorPlanetMassType(RV_epochs[RVdataSet][epoch], e_p, T_p, p_p, argPeri_p, M1=0, T_center=Tc_p, M2SineI=False, K=K_p, verbose=False)
