479 def burnInCalcFunc(chiSquareds, medianALLchains,jumpy=True):
480  """
481  This function will calculate the burn in length and return its value.
482  This can only truly be done for a proper MCMC that was started
483  at a random location in the parameter space, as anything else will
484  skew the chiSquareds trend. ie. they will not roughly decrease as the
485  simulation runs and the burn in length will thus be more or less random and
486  useless information.
487  Burn-in is defined as the first point in a chain when the chi squared is equal or less
488  than the median value of all the chains in the simulation. Thus, there MUST
489  be more than 1 chain to perform this calculation.
490 
491  :param paramIN: parameter array including burn in data
492  :type paramIN: array (list) of doubles
493  """
494  burnInLength = 0
495  verbose = False
496 
497  if jumpy:
498  if len(chiSquareds)>=100000:
499  jump = 100
500  else:
501  jump = 10
502  else:
503  jump = 1
504 
505  if verbose:
506  print 'jump size = '+str(jump)
507  print 'medianALLchains = '+str(medianALLchains)
508 
509  for i in range(0,int(len(chiSquareds)/jump)):
510  # check if current chiSquared is less than the median of all chains yet
511  chiSquareCur = chiSquareds[i*jump]
512  if chiSquareCur<medianALLchains:
513  if verbose:
514  print 'found '+str(chiSquareCur)+' < '+str(medianALLchains)+' at step '+str(i*jump)+', so going back to previous jump step and finding exact location'
515  if jump>1:
516  # less than median, so do all in last jump to find precise location
517  for j in range((i-1)*jump, i*jump):
518  chiSquareCur2 = chiSquareds[j]
519  if chiSquareCur2<medianALLchains:
520  burnInLength = j+1
521  if verbose:
522  print 'found '+str(chiSquareCur2)+' < '+str(medianALLchains)+' at step '+str(j)+', so returning burnInLength as '+str(burnInLength)
523  break
524  else:
525  burnInLength = i+1
526  break
527  if burnInLength == len(chiSquareds):
528  print "PROBLEM: Param had a burn in length equal to param length, ie. the chain never burned in"
529 
530  return burnInLength

1009 def effectivePointsCalcFunc(data):
1010  """
1011  for use by MCMCeffectivePointsCalc
1012  """
1013  # Important: if your chain is 1 million steps, then ecc should have
1014  # 1 million elements. Don't have a separate array indicating how
1015  # many times you visited each point. N_pts is the total number of

1020

1023  var_obs = np.var(data)

1040
