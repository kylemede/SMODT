50 void SplitIntoGenotypes(int32_t * genotypes, int32_t * genes, StochasticLib1 & stoc) {
51  // split gene pool into (a,a), (a,b) and (b,b) genotypes
52  // parameters:
53  // genotypes: output. array with 3 places.
54  // gets number of (a,a), (a,b), and (b,b) genotypes, respectively
55  // genes: gene pool. array with 2 places
56  // stoc: random library
57 
58  int32_t g; // total number of genes
59  int32_t na_; // number of individuals with gene a on first place
60 
61  g = genes[0] + genes[1];
62  na_ = stoc.Hypergeometric(g/2, genes[0], g);
63  genotypes[0] = stoc.Hypergeometric(na_, genes[0] - na_, g/2);
64  genotypes[1] = genes[0] - 2*genotypes[0];
65  genotypes[2] = g/2 - genotypes[0] - genotypes[1];}

98  int32_t GenerationNumber; // generation number

100  int32_t Children; // number of individuals in child population
101  int32_t MaxPopulation; // maximum population size
102  int32_t Genotypes[3]; // genotypes (a,a), (a,b), and (b,b)
103  double Fitness[3]; // relative fitness for each genotype
104  int32_t OutputPeriod; // period between outputs
105 
106  //---------------------------------------------------------------------------
107  // initializations
108  //---------------------------------------------------------------------------
109  // Calculate reciprocal relative fitness of each genotype
110  Fitness[0] = 1.;
111  Fitness[1] = 1. + SelectionCoefficient * Dominance;
112  Fitness[2] = 1. + SelectionCoefficient;
113 
114  // Make initial gene pool
115  ParentGenePool[0] = 2 * PopulationSize;
116  ParentGenePool[1] = 0;
117 
118  // Prepare output
119  OutputPeriod = Generations / 20;
120  printf("generation gene a gene b");

122 
123  //---------------------------------------------------------------------------
124  // Generation loop
125  //---------------------------------------------------------------------------
126  for (GenerationNumber = 1; GenerationNumber <= Generations; GenerationNumber++) {
127 
128  // Mutation
129  Mutations = sto.Binomial(ParentGenePool[0], MutationRate)
130  - sto.Binomial(ParentGenePool[1], MutationRate);
131  ParentGenePool[0] -= Mutations;
132  ParentGenePool[1] += Mutations;
133 
134  // Breeding
135  // Number of children can be a poisson or normal distribution
136  Children = sto.Poisson(PopulationSize * GrowthRate);
137 
138  // Child gene pool follows binomial distribution
139  sto.Multinomial (ChildGenePool, ParentGenePool, Children*2, 2);
140 
141  // Limit population size
142  PopulationSize = Children;
143  MaxPopulation = (int32_t)sto.Normal(CarryingCapacity, CapacityDeviation);
144  if (MaxPopulation < 0) MaxPopulation = 0;
145  if (PopulationSize > MaxPopulation) PopulationSize = MaxPopulation;
146 
147  // Split child gene pool into (a,a), (a,b) and (b,b) genotypes
148  SplitIntoGenotypes(Genotypes, ChildGenePool, sto);
149

152  // noncentral hypergeometric distribution. The number of survivals follows

155

158  ChildGenePool[1] = Genotypes[1] + Genotypes[2]*2;
159 
160  // Generation shift
161  ParentGenePool[0] = ChildGenePool[0];
162  ParentGenePool[1] = ChildGenePool[1];
163 
164  // Output

168  if (PopulationSize == 0) break;}
